56  /**
57   * <p>
58   * An webservice meant to be used in tests to simulate a HTTP Server. You can feed
59   * it with prepared responses which will then be sent on each request.
60   * </p>
61   *
62   * @deprecated Will be removed soon.
63   */
64  @SuppressWarnings("deprecation")
65  @Deprecated
66  public class EmbeddedWebservice {
67    /**
68     * Logging instance.
69     */
70    private static final Logger LOG = LoggerFactory.getLogger(EmbeddedWebservice.class);
71    private static final TimeoutImpl SERVER_STARTUP_TIMEOUT = new TimeoutImpl(5L, TimeUnit.SECONDS);
72  
73    /**
74     * Server to bind to port.
75     */
76    private volatile HttpServer server;
77    /**
78     * Http Handler which will provide prepared responses on each request.
79     */
80    @Nonnull
81    private final PreparedResponsesHttpHandler preparedResponsesHttpHandler;
82    /**
83     * The URI you can send requests to, which will reach this webservice.
84     */
85    @Nonnull
86    private final URI clientUri;
87    /**
88     * Context the service is bound to.
89     */
90    @Nonnull
91    private final String context;
92    /**
93     * Port the service is bound to.
94     */
95    private final int port;
96  
97    /**
98     * <p>
99     * Creates a webservice responding to a random free port on root context.
100    * </p>
101    *
102    * @throws IOException on failure
103    */
104   public EmbeddedWebservice() throws IOException {
105     this("/");
106   }
107 
108   /**
109    * <p>
110    * Creates a webservice responding to a random free port on specified context.
111    * </p>
112    *
113    * @param context context, for example {@code /lorem} for lorem-context
114    * @throws IOException on failure
115    */
116   public EmbeddedWebservice(@Nonnull final String context) throws IOException {
117     this(context, freePort());
118   }
119 
120   /**
121    * <p>
122    * Creates a webservice responding to given port on specified context.
123    * </p>
124    *
125    * @param context context, for example {@code /lorem} for lorem-context
126    * @param port    port to bind the service to
127    * @throws IOException on failure
128    */
129   public EmbeddedWebservice(@Nonnull final String context, final int port) throws IOException {
130     checkNotNull(context, "Context must not be null. For root context use '/'.");
131     try {
132       this.context = context;
133       this.port = port;
134       final InetSocketAddress address = new InetSocketAddress(port);
135       server = HttpServer.create(address, 0);
136       preparedResponsesHttpHandler = new PreparedResponsesHttpHandlerImpl();
137       server.createContext(this.context, preparedResponsesHttpHandler);
138       server.setExecutor(null);
139       clientUri = URI.create(format("http://%s:%d%s", getByName(null).getHostName(), this.port, context));
140     } catch (IOException e) {
141       throw new IOException("Failed to prepare embedded webservice for " + context + ':' + port, e);
142     }
143     LOG.info("Embedded Webservice ready to start for " + context + ':' + port);
144   }
145 
146   /**
147    * <p>
148    * Get the URI which where to send requests to, to reach this service.
149    * </p>
150    *
151    * @return client URI
152    */
153   @Nonnull
154   public URI getClientUri() {
155     return clientUri;
156   }
157 
158   /**
159    * Start the webservice.
160    */
161   public void start() {
162     checkState(server != null, "Server cannot be restarted.");
163     server.start();
164     preparedResponsesHttpHandler.feedResponses(statusCode(HttpURLConnection.HTTP_OK));
165     final DefaultHttpClient httpClient = getStartupHttpClient();
166     try {
167       getStartupCondition(httpClient).assumeThat(equalTo(HttpURLConnection.HTTP_OK));
168     } finally {
169       httpClient.getConnectionManager().shutdown();
170     }
171     LOG.info("Started embedded webservice at port {} with context {}.", port, context);
172   }
173 
174   private Condition<Integer> getStartupCondition(final DefaultHttpClient httpClient) {
175     return new DefaultConditionFactory(SERVER_STARTUP_TIMEOUT).condition(new ServerStartupStatusCodeExpression(httpClient));
176   }
177 
178   private DefaultHttpClient getStartupHttpClient() {
179     final DefaultHttpClient httpClient = new DefaultHttpClient();
180     final HttpParams httpParams = httpClient.getParams();
181     HttpConnectionParams.setConnectionTimeout(httpParams, (int) SERVER_STARTUP_TIMEOUT.in(TimeUnit.MILLISECONDS));
182     HttpConnectionParams.setSoTimeout(httpParams, (int) SERVER_STARTUP_TIMEOUT.in(TimeUnit.MILLISECONDS));
183     return httpClient;
184   }
185 
186   /**
187    * Stop the webservice.
188    */
189   public void stop() {
190     server.stop(1);
191     preparedResponsesHttpHandler.clearResponses();
192     server = null;
193     LOG.info("Stopped embedded webservice");
194   }
195 
196   /**
197    * <p>
198    * Get handler to control response flow.
199    * </p>
200    *
201    * @return handler to provide responses
202    */
203   @Nonnull
204   public PreparedResponsesHttpHandler getHttpHandler() {
205     return preparedResponsesHttpHandler;
206   }
207 
208   @Override
209   public String toString() {
210     return MoreObjects.toStringHelper(this)
211             .add("clientUri", clientUri)
212             .add("server", server)
213             .add("preparedResponsesHttpHandler", preparedResponsesHttpHandler)
214             .add("context", context)
215             .add("port", port)
216             .toString();
217   }
218 
219   private class ServerStartupStatusCodeExpression extends AbstractExpression<Integer> {
220     private final DefaultHttpClient httpClient;
221 
222     private ServerStartupStatusCodeExpression(final DefaultHttpClient httpClient) {
223       this.httpClient = httpClient;
224     }
225 
226     @Override
227     public Integer get() {
228       final HttpUriRequest httpHead = new HttpHead(clientUri);
229       try {
230         final HttpResponse response = httpClient.execute(httpHead);
231         final HttpEntity httpEntity = response.getEntity();
232         final StatusLine statusLine = response.getStatusLine();
233         final int statusCode = statusLine.getStatusCode();
234         if (httpEntity != null) {
235           EntityUtils.consume(httpEntity);
236         }
237         return statusCode;
238       } catch (IOException e) {
239         throw new ExpressionEvaluationException(format("Failure reading from URI %s.", clientUri), e);
240       }
241     }
242   }
