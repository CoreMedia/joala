20  package net.joala.condition;
21  
22  import net.joala.condition.timing.WaitTimeoutException;
23  import net.joala.expression.Expression;
24  import net.joala.expression.ExpressionEvaluationException;
25  import org.hamcrest.Matcher;
26  
27  import javax.annotation.Nonnegative;
28  import javax.annotation.Nonnull;
29  import javax.annotation.Nullable;
30  
31  /**
32   * A test condition to wait for using a given timeout. If the condition does not become true within
33   * time exceptions will be thrown.
34   *
35   * @param <T> the type of the value which will be verified
36   * @since 2/24/12
37   */
38  public interface Condition<T> {
39    /**
40     * Conditions need to implement this method to return the value
41     * which will be checked.
42     *
43     * @return the value
44     * @throws ExpressionEvaluationException when the actual value cannot be evaluated. Consider calling any of the
45     *                                       {@code await} or {@code assume} methods to wait for the value to
46     *                                       become available.
47     * @see Expression#get()
48     */
49    T get();
50  
51    /**
52     * <p>
53     * Retrieve the result of the condition as soon as it is evaluated without any exception.
54     * </p>
55     *
56     * @return the retrieved value
57     * @throws WaitTimeoutException if a value could not be retrieved in time
58     */
59    T await();
60  
61    /**
62     * <p>
63     * Wait for the condition value until it fulfills the given matcher and return it.
64     * </p>
65     *
66     * @param matcher matcher to use
67     * @return the value which fulfills the given matcher
68     * @throws WaitTimeoutException if a value could not be retrieved in time
69     */
70    T await(@Nonnull Matcher<? super T> matcher);
71  
72    /**
73     * <p>
74     * Assumes that the condition evaluates to the expected value within a given time.
75     * </p>
76     *
77     * @param expected expected value
78     */
79    void assumeEquals(@Nullable T expected);
80  
81    /**
82     * Uses {@link Matcher} to assume that a condition is met in time.
83     *
84     * @param matcher the matcher to use
85     */
86    void assumeThat(@Nonnull Matcher<? super T> matcher);
87  
88    /**
89     * <p>
90     * Assert that the condition evaluates to the expected value within a given time.
91     * </p>
92     *
93     * @param expected expected value
94     */
95    void assertEquals(@Nullable T expected);
96  
97    /**
98     * Uses {@link Matcher} to assert that a condition is met in time.
99     *
100    * @param matcher the matcher to use
101    */
102   void assertThat(@Nonnull Matcher<? super T> matcher);
103 
104   /**
105    * <p>
106    * Wait until the condition evaluates to the expected value within a given time.
107    * </p>
108    *
109    * @param expected expected value
110    */
111   void waitUntilEquals(@Nullable T expected);
112 
113   /**
114    * Uses {@link Matcher} to wait until a condition is met in time.
115    *
116    * @param matcher the matcher to use
117    */
118   void waitUntil(@Nonnull Matcher<? super T> matcher);
119 
120   /**
121    * Sets a factor to speed up or slow down tests. If &lt; 1.0 the tests will speed up at least
122    * for failures while a number &gt; 1.0 increases the time to wait for the condition by
123    * the given factor.
124    *
125    * @param factor factor by which to increase the timeout for this condition
126    * @return a self-reference.
127    */
128   @Nonnull
129   Condition<T> withTimeoutFactor(@Nonnegative double factor);
130 
131   /**
132    * The message to print on failure.
133    *
134    * @param message message with debugging hints; {@code null} to remove the message
135    * @return self-reference
136    */
137   @Nonnull
138   Condition<T> withMessage(@Nullable String message);
139 }
