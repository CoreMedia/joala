20  package net.joala.testlet;
21  
22  import com.google.common.base.MoreObjects;
23  import org.junit.runner.notification.Failure;
24  import org.junit.runner.notification.RunListener;
25  import org.junit.runners.model.MultipleFailureException;
26  
27  import java.util.ArrayList;
28  import java.util.List;
29  
30  /**
31   * <p>
32   *   Listener while running the testlets which will collect any failures.
33   * </p>
34   * @since 10/4/12
35   */
36  class ExceptionCollectingRunListener extends RunListener {
37    /**
38     * Assumption on number of test failures.
39     */
40    private static final int INITIALLY_EXPECTED_FAILURE_COUNT = 5;
41    /**
42     * List of collected exceptions.
43     */
44    private final List<Throwable> exceptions = new ArrayList<Throwable>(INITIALLY_EXPECTED_FAILURE_COUNT);
45  
46    @Override
47    public void testFailure(final Failure failure) {
48      exceptions.add(failure.getException());
49    }
50  
51    @Override
52    public void testAssumptionFailure(final Failure failure) {
53      exceptions.add(failure.getException());
54    }
55  
56    /**
57     * <p>
58     * Asserts that no exceptions occurred. See {@link MultipleFailureException#assertEmpty(List)}
59     * for details.
60     * </p>
61     *
62     * @throws Throwable raised by {@link MultipleFailureException#assertEmpty(List)} if any errors got reported
63     */
64    @SuppressWarnings("ProhibitedExceptionDeclared")
65    public void assertNoFailures() throws Throwable { // NOSONAR: exception Throwable inherited from JUnit API
66      MultipleFailureException.assertEmpty(exceptions);
67    }
68  
69    @Override
70    public String toString() {
71      return MoreObjects.toStringHelper(this)
72              .add("exceptions", exceptions)
73              .toString();
74    }
75  }
