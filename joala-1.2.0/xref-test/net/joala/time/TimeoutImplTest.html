20  package net.joala.time;
21  
22  import org.junit.Test;
23  
24  import java.util.PrimitiveIterator;
25  import java.util.Random;
26  import java.util.concurrent.TimeUnit;
27  import java.util.function.Supplier;
28  
29  import static java.lang.Math.round;
30  import static java.lang.String.format;
31  import static net.joala.testlet.ToStringTestlet.toStringTestlet;
32  import static org.junit.Assert.assertEquals;
33  
34  /**
35   * Tests {@link TimeoutImpl}.
36   *
37   * @since 8/25/12
38   */
39  @SuppressWarnings("ProhibitedExceptionDeclared")
40  public class TimeoutImplTest {
41    private static final PrimitiveIterator.OfInt INTS = new Random().ints(0, Integer.MAX_VALUE).iterator();
42    private final Supplier<Integer> randomPositiveInt = INTS::next;
43    private static final PrimitiveIterator.OfDouble DOUBLES = new Random().doubles(0D, Double.MAX_VALUE).iterator();
44    private final Supplier<Double> randomPositiveDouble = DOUBLES::next;
45  
46    @Test(expected = IllegalArgumentException.class)
47    public void constructor_should_throw_exception_on_negative_timeout() throws Exception {
48      new TimeoutImpl(-1L, TimeUnit.MILLISECONDS);
49    }
50  
51    @Test
52    public void constructor_should_set_amount_and_unit_correctly() throws Exception {
53      final int amount = randomPositiveInt.get();
54      final TimeUnit unit = TimeUnit.SECONDS;
55      final Timeout timeout = new TimeoutImpl(amount, unit);
56      assertEquals("Constructor should have correctly set amount and unit.", unit.toMillis(amount), timeout.in(TimeUnit.MILLISECONDS));
57    }
58  
59    @Test
60    public void in_method_should_convert_correctly() throws Exception {
61      final int amount = randomPositiveInt.get();
62      final TimeUnit unit = TimeUnit.SECONDS;
63      final Timeout timeout = new TimeoutImpl(amount, unit);
64      assertEquals(format("Correctly converted to Milliseconds: %s", timeout), unit.toMillis(amount), timeout.in(TimeUnit.MILLISECONDS));
65      assertEquals(format("Correctly converted to Seconds: %s", timeout), unit.toSeconds(amount), timeout.in(TimeUnit.SECONDS));
66      assertEquals(format("Correctly converted to Minutes: %s", timeout), unit.toMinutes(amount), timeout.in(TimeUnit.MINUTES));
67    }
68  
69    @Test
70    public void in_method_should_correctly_apply_positive_factor() throws Exception {
71      final double factor = randomPositiveDouble.get();
72      final int amount = randomPositiveInt.get();
73      final TimeUnit unit = TimeUnit.SECONDS;
74      final Timeout timeout = new TimeoutImpl(amount, unit);
75      assertEquals(format("Correctly converted to Milliseconds: %s", timeout), round(unit.toMillis(amount) * factor), timeout.in(TimeUnit.MILLISECONDS, factor));
76      assertEquals(format("Correctly converted to Seconds: %s", timeout), round(unit.toSeconds(amount) * factor), timeout.in(TimeUnit.SECONDS, factor));
77    }
78  
79    @Test(expected = IllegalArgumentException.class)
80    public void in_method_should_throw_exception_on_non_positive_factor() throws Exception {
81      final double factor = -1.0 * randomPositiveDouble.get();
82      final int amount = randomPositiveInt.get();
83      final TimeUnit unit = TimeUnit.SECONDS;
84      final Timeout timeout = new TimeoutImpl(amount, unit);
85      timeout.in(unit, factor);
86    }
87  
88    @Test
89    public void toString_should_contain_necessary_information() throws Throwable {
90      final int amount = randomPositiveInt.get();
91      final TimeUnit unit = TimeUnit.SECONDS;
92      final Timeout timeout = new TimeoutImpl(amount, unit);
93      toStringTestlet(timeout).run();
94    }
95  
96  }
