20  package net.joala.matcher.net;
21  
22  import org.hamcrest.CustomTypeSafeMatcher;
23  import org.hamcrest.Factory;
24  import org.hamcrest.Matcher;
25  
26  import java.net.UnknownHostException;
27  
28  import static java.net.InetAddress.getAllByName;
29  
30  /**
31   * <p>
32   * Validates hostnames if they can be resolved, thus are known to the system.
33   * </p>
34   *
35   * @since 10/2/12
36   */
37  public class KnownHost extends CustomTypeSafeMatcher<String> {
38    public KnownHost() {
39      super("hostname which can be resolved to IP address");
40    }
41  
42    @Override
43    protected boolean matchesSafely(final String item) {
44      try {
45        //noinspection ResultOfMethodCallIgnored
46        getAllByName(item);
47      } catch (UnknownHostException ignored) {
48        return false;
49      }
50      return true;
51    }
52  
53    /**
54     * <p>
55     * Validates hostnames if they can be resolved, thus are known to the system.
56     * </p>
57     *
58     * @return hostname validating matcher
59     */
60    @Factory
61    public static Matcher<String> knownHost() {
62      return new KnownHost();
63    }
64  
65  }
