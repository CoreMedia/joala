20  package net.joala.bdd.reference;
21  
22  import javax.annotation.Nonnull;
23  import javax.annotation.Nullable;
24  
25  /**
26   * <p>
27   * A reference is meant to be used to be shared among steps of BDD tests written in JUnit. It is typically
28   * created before a test, filled once with a value during the tests and then can be read multiple times.
29   * </p>
30   * <p>
31   * References can hold properties that can be used for later assertions.
32   * </p>
33   *
34   * @param <T> the type of the reference value
35   * @see net.joala.bdd
36   * @since 6/5/12
37   */
38  public interface Reference<T> {
39    /**
40     * <p>
41     * Set reference value. Can only be called once.
42     * </p>
43     *
44     * @param value the value of the reference
45     * @throws ReferenceAlreadyBoundException if the reference already has a value
46     */
47    void set(@Nullable T value);
48  
49    /**
50     * <p>
51     * Retrieve the value of a reference.
52     * </p>
53     *
54     * @return the value of the reference
55     * @throws ReferenceNotBoundException if the reference does not contain a value yet
56     * @see #hasValue()
57     */
58    @Nullable
59    T get();
60  
61    /**
62     * Signals if the reference carries a value or not.
63     *
64     * @return {@code true} if reference carries a value, {@code false} if not
65     * @since 1.1.0
66     */
67    boolean hasValue();
68  
69    /**
70     * <p>
71     * Set a property which belongs to the reference value. Such a property might for example be a text which will
72     * be represent the reference value in a webapp. For example a username or something similar.
73     * </p>
74     *
75     * @param key   the name of the property
76     * @param value the value of the property
77     * @throws NullPointerException        if the key is {@code null}
78     * @throws PropertyAlreadySetException if you already set this property
79     */
80    void setProperty(@Nonnull String key, @Nullable Object value);
81  
82    /**
83     * <p>
84     * Read the property value.
85     * </p>
86     *
87     * @param key the name of the property
88     * @return property value
89     * @throws PropertyNotSetException if you did not define that property before
90     */
91    @Nullable
92    Object getProperty(@Nonnull String key);
93  
94    /**
95     * <p>
96     * Read the property value.
97     * </p>
98     *
99     * @param key   the name of the property
100    * @param clazz the type of the property value to cast it to
101    * @param <P>   the type of the property value
102    * @return property value
103    * @throws PropertyNotSetException if you did not define that property before
104    */
105   @Nullable
106   <P> P getProperty(@Nonnull String key, @Nonnull Class<P> clazz);
107 
108   /**
109    * <p>
110    * Checks, whether a property with the given key is set.
111    * </p>
112    *
113    * @param key the name of the property
114    * @return {@code true} if the property is set, otherwise {@code false}
115    * @since 1.1.0
116    */
117   boolean hasProperty(@Nonnull String key);
118 
119   /**
120    * Removes the property with the given key.
121    *
122    * @param key           the name of the property
123    * @param expectedClass the type of the removed property value to cast it to
124    * @param <P>           the type of the property value
125    * @return former property value
126    * @throws PropertyNotSetException if you did not define that property before
127    * @since 1.1.0
128    */
129   @Nullable
130   <P> P removeProperty(@Nonnull String key, @Nonnull Class<P> expectedClass);
131 
132   /**
133    * Removes the property with the given key.
134    *
135    * @param key the name of the property
136    * @return former property value
137    * @throws PropertyNotSetException if you did not define that property before
138    * @since 1.1.0
139    */
140   Object removeProperty(@Nonnull String key);
141 
142 }
