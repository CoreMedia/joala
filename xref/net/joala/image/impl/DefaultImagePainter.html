20  package net.joala.image.impl;
21  
22  import java.awt.Color;
23  import java.awt.GradientPaint;
24  import java.awt.Graphics2D;
25  import java.awt.Paint;
26  import java.awt.image.BufferedImage;
27  
28  /**
29   * Default image painter drawing some unspecified image.
30   *
31   * @since 2013-03-11
32   */
33  public class DefaultImagePainter extends AbstractImagePainter {
34    @Override
35    protected void paint(final BufferedImage image, final Graphics2D g2d) {
36      final int width = image.getWidth();
37      final int height = image.getHeight();
38      final Paint gradientWhiteBlue = new GradientPaint(0, 0, Color.WHITE, width, 0, Color.BLUE);
39      final Paint gradientBlueWhite = new GradientPaint(0, 0, Color.BLUE, width, 0, Color.WHITE);
40      g2d.setPaint(gradientWhiteBlue);
41      g2d.fillRect(0, 0, width, height);
42      g2d.setPaint(gradientBlueWhite);
43      g2d.fillOval(0, 0, width, height);
44    }
45  }
