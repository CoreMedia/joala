20  package net.joala.image.impl;
21  
22  import net.joala.image.ByteArrayImageBuilder;
23  import net.joala.image.OutputImageBuilder;
24  import net.joala.image.config.ImageType;
25  
26  import javax.activation.MimeType;
27  import javax.annotation.CheckForNull;
28  import javax.annotation.Nonnull;
29  import javax.inject.Inject;
30  import javax.inject.Named;
31  import java.io.ByteArrayOutputStream;
32  
33  import static com.google.common.base.Optional.fromNullable;
34  
35  /**
36   * <p>
37   * Creates an image as byte array.
38   * </p>
39   *
40   * @since 2013-02-19
41   */
42  @Named
43  public class DefaultByteArrayImageBuilder extends AbstractImageIOImageBuilder implements ByteArrayImageBuilder {
44    @Inject
45    @CheckForNull
46    private OutputImageBuilder outputImageBuilder;
47  
48    @Override
49    public ByteArrayImageBuilder width(final int value) {
50      setWidth(value);
51      return this;
52    }
53  
54    @Override
55    public ByteArrayImageBuilder height(final int value) {
56      setHeight(value);
57      return this;
58    }
59  
60    @Override
61    public ByteArrayImageBuilder mimeType(final MimeType value) {
62      setMimeType(value);
63      return this;
64    }
65  
66    @Override
67    public ByteArrayImageBuilder mimeType(final String value) {
68      setMimeType(value);
69      return this;
70    }
71  
72    @Override
73    public ByteArrayImageBuilder imageType(final ImageType value) {
74      setImageType(value);
75      return this;
76    }
77  
78    @SuppressWarnings("TypeMayBeWeakened") // NOSONAR
79    @Nonnull
80    private OutputImageBuilder getOutputImageBuilder() {
81      return fromNullable(outputImageBuilder).or(new DefaultOutputImageBuilder());
82    }
83  
84    @Override
85    public byte[] build() {
86      final ByteArrayOutputStream output = new ByteArrayOutputStream();
87      getOutputImageBuilder().width(getWidth())
88              .height(getHeight())
89              .mimeType(getMimeType())
90              .imageType(getImageType())
91              .output(output)
92              .build();
93      return output.toByteArray();
94    }
95  
96  }
