20  package net.joala.condition;
21  
22  import net.joala.expression.Expression;
23  
24  import javax.annotation.Nonnull;
25  
26  /**
27   * <p>
28   * Interface for factories providing conditions.
29   * </p>
30   *
31   * @since 8/22/12
32   */
33  public interface ConditionFactory {
34    /**
35     * <p>
36     * Create a boolean condition which allows some convenience methods for assertions and assumptions.
37     * </p>
38     *
39     * @param expression the expression which will evaluate to a boolean value
40     * @return boolean condition
41     * @throws NullPointerException if expression is null
42     */
43    @Nonnull
44    BooleanCondition booleanCondition(@Nonnull Expression<Boolean> expression);
45  
46    /**
47     * <p>
48     * Create a ready to use condition.
49     * </p>
50     *
51     * @param expression the expression which will evaluate to value of type T
52     * @param <T>        the type the expression value will have
53     * @return condition
54     * @throws NullPointerException if expression is null
55     */
56    @Nonnull
57    <T> Condition<T> condition(@Nonnull Expression<T> expression);
58  }
