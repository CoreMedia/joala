27  import org.apache.http.HttpResponse;
28  import org.apache.http.StatusLine;
29  import org.apache.http.client.methods.HttpHead;
30  import org.apache.http.client.methods.HttpUriRequest;
31  import org.apache.http.impl.client.DefaultHttpClient;
32  import org.apache.http.params.HttpConnectionParams;
33  import org.apache.http.params.HttpParams;
34  import org.apache.http.util.EntityUtils;
35  
36  import javax.annotation.Nonnull;
37  import java.io.IOException;
38  import java.net.URI;
39  import java.util.concurrent.TimeUnit;
40  
41  import static com.google.common.base.Preconditions.checkState;
42  import static java.lang.String.format;
43  import static net.joala.matcher.net.KnownHost.knownHost;
44  
45  /**
46   * <p>
47   * Matcher to validate response code received from given URI.
48   * </p>
49   *
50   * @deprecated Will be removed soon.
51   */
52  @SuppressWarnings({"deprecation", "IOResourceOpenedButNotSafelyClosed"})
53  @Deprecated
54  public class UriStatusCodeExpression extends AbstractExpression<Integer> {
55    /**
56     * Default timeout to use if unspecified.
57     */
58    private static final Timeout DEFAULT_TIMEOUT = new TimeoutImpl(10, TimeUnit.SECONDS);
59  
60    /**
61     * Timeout for connection.
62     */
63    private final Timeout timeout;
64    private final URI uri;
65  
66    /**
67     * <p>
68     * Constructor with target URI. Default timeout is used.
69     * </p>
70     *
71     * @param uri URI to get the status code from
72     */
73    public UriStatusCodeExpression(@Nonnull final URI uri) {
74      this(DEFAULT_TIMEOUT, uri);
75    }
76  
77    /**
78     * <p>
79     * Constructor with target URI and connection timeout.
80     * </p>
81     *
82     * @param timeout the connection timeout
83     * @param uri     URI to get the status code from
84     */
85    public UriStatusCodeExpression(@Nonnull final Timeout timeout, @Nonnull final URI uri) {
86      super(format("Retrieve status code from URI %s", uri));
87      this.timeout = timeout;
88      this.uri = uri;
89    }
90  
91    @Override
92    @Nonnull
93    public Integer get() {
94      final String host = uri.getHost();
95      checkState(knownHost().matches(host), "Host %s from URI %s is unknown.", host, uri);
96      final DefaultHttpClient httpClient = new DefaultHttpClient();
97      try {
98        final HttpParams httpParams = httpClient.getParams();
99        HttpConnectionParams.setConnectionTimeout(httpParams, (int) timeout.in(TimeUnit.MILLISECONDS));
100       HttpConnectionParams.setSoTimeout(httpParams, (int) timeout.in(TimeUnit.MILLISECONDS));
101       final HttpUriRequest httpHead = new HttpHead(uri);
102       try {
103         final HttpResponse response = httpClient.execute(httpHead);
104         final HttpEntity httpEntity = response.getEntity();
105         final StatusLine statusLine = response.getStatusLine();
106         final int statusCode = statusLine.getStatusCode();
107         if (httpEntity != null) {
108           EntityUtils.consume(httpEntity);
109         }
110         return statusCode;
111       } catch (IOException e) {
112         throw new ExpressionEvaluationException(format("Failure reading from URI %s.", uri), e);
113       }
114     } finally {
115       httpClient.getConnectionManager().shutdown();
116     }
117   }
