20  package net.joala.condition.timing;
21  
22  import com.google.common.base.Function;
23  import org.hamcrest.Description;
24  import org.hamcrest.SelfDescribing;
25  import org.hamcrest.StringDescription;
26  
27  import javax.annotation.Nonnull;
28  import javax.annotation.Nullable;
29  
30  import static com.google.common.base.Preconditions.checkNotNull;
31  import static net.joala.matcher.DescriptionUtil.describeTo;
32  
33  /**
34   * <p>
35   * Exception which evaluated functions might throw in order to signal that the current state query failed but should
36   * be retried.
37   * </p>
38   * <p>
39   * If the function implements {@link SelfDescribing} its description will be added to the failure message
40   * or taken as failure message if no extra message is provided. Otherwise its {@link Object#toString()} method
41   * will be called.
42   * </p>
43   *
44   * @see Wait
45   * @since 9/14/12
46   */
47  public class IgnorableStateQueryException extends RuntimeException {
48    private static final String CHECK_QUERY_FUNCTION_MUST_NOT_BE_NULL = "Query function must not be null";
49    @Nonnull
50    private final Function<?, ?> stateQuery;
51  
52    /**
53     * <p>
54     * Constructor with the query which raised the error.
55     * </p>
56     *
57     * @param stateQuery query raising this exception
58     */
59    public IgnorableStateQueryException(@Nonnull final Function<?, ?> stateQuery) {
60      super();
61      checkNotNull(stateQuery, CHECK_QUERY_FUNCTION_MUST_NOT_BE_NULL);
62      this.stateQuery = stateQuery;
63    }
64  
65    /**
66     * <p>
67     * Constructor with query and cause.
68     * </p>
69     *
70     * @param stateQuery query raising this exception
71     * @param cause      exception which caused the failure on state query
72     */
73    public IgnorableStateQueryException(@Nonnull final Function<?, ?> stateQuery, final Throwable cause) {
74      super(cause);
75      checkNotNull(stateQuery, CHECK_QUERY_FUNCTION_MUST_NOT_BE_NULL);
76      this.stateQuery = stateQuery;
77    }
78  
79    /**
80     * <p>
81     * Constructor with query and message.
82     * </p>
83     *
84     * @param stateQuery query raising this exception
85     * @param message    message for this failure, will be enriched by the description of the stateQuery
86     */
87    public IgnorableStateQueryException(@Nonnull final Function<?, ?> stateQuery, final String message) {
88      super(message);
89      checkNotNull(stateQuery, CHECK_QUERY_FUNCTION_MUST_NOT_BE_NULL);
90      this.stateQuery = stateQuery;
91    }
92  
93    /**
94     * <p>
95     * Constructor with query, message and cause.
96     * </p>
97     *
98     * @param stateQuery query raising this exception
99     * @param message    message for this failure, will be enriched by the description of the stateQuery
100    * @param cause      exception which caused the failure on state query
101    */
102   public IgnorableStateQueryException(@Nonnull final Function<?, ?> stateQuery, final String message, final Throwable cause) {
103     super(message, cause);
104     checkNotNull(stateQuery, CHECK_QUERY_FUNCTION_MUST_NOT_BE_NULL);
105     this.stateQuery = stateQuery;
106   }
107 
108   @Override
109   public String getMessage() {
110     return buildMessage(stateQuery, super.getMessage());
111   }
112 
113   @Override
114   public String getLocalizedMessage() {
115     return buildMessage(stateQuery, super.getLocalizedMessage());
116   }
117 
118   /**
119    * <p>
120    * Build a message combining the original message (if any) with a description of the stateQuery.
121    * </p>
122    *
123    * @param function function which queried the state
124    * @param message  message provided for exception; might be {@code null}
125    * @return a complete message for the exception
126    */
127   private static String buildMessage(@Nonnull final Function<?, ?> function, @Nullable final String message) {
128     final Description description = new StringDescription();
129     if (message == null) {
130       description.appendText("Failure: ");
131     } else {
132       description.appendText(message);
133       description.appendText(" - ");
134     }
135     describeTo(description, function);
136     return description.toString();
137   }
138 
139 }
