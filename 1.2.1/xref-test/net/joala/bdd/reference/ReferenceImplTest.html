20  package net.joala.bdd.reference;
21  
22  import org.junit.Test;
23  
24  import java.util.Collections;
25  import java.util.Map;
26  
27  import static org.junit.Assert.assertEquals;
28  import static org.junit.Assert.assertFalse;
29  import static org.junit.Assert.assertNull;
30  import static org.junit.Assert.assertSame;
31  import static org.junit.Assert.assertTrue;
32  
33  /**
34   * <p>
35   * Test for {@link ReferenceImpl}.
36   * </p>
37   *
38   * @since 6/5/12
39   */
40  @SuppressWarnings("ProhibitedExceptionDeclared")
41  public class ReferenceImplTest {
42    @Test
43    public void should_hold_value() throws Exception {
44      final Reference<Map<Object, Object>> reference = new ReferenceImpl<Map<Object, Object>>();
45      final Map<Object, Object> referenceValue = Collections.emptyMap();
46      reference.set(referenceValue);
47      assertSame("Reference value retrieved should hold the original value set.", referenceValue, reference.get());
48    }
49  
50    @Test(expected = ReferenceAlreadyBoundException.class)
51    public void should_deny_to_set_value_twice() throws Exception {
52      final Reference<String> reference = new ReferenceImpl<String>();
53      reference.set("Lorem");
54      reference.set("Ipsum");
55    }
56  
57    @Test(expected = ReferenceNotBoundException.class)
58    public void should_deny_to_read_from_unbound_reference() throws Exception {
59      final Reference<String> reference = new ReferenceImpl<String>();
60      reference.get();
61    }
62  
63    @Test
64    public void should_be_able_to_hold_reference_value_null() throws Exception {
65      final Reference<String> reference = new ReferenceImpl<String>();
66      reference.set(null);
67      assertNull("Value null should be returned when retrieving value.", reference.get());
68    }
69  
70    @Test
71    public void should_hold_property_value() throws Exception {
72      final String propertyKey = "lorem";
73      final Map<Object, Object> propertyValue = Collections.emptyMap();
74      final Reference<String> reference = new ReferenceImpl<String>();
75      reference.setProperty(propertyKey, propertyValue);
76      assertSame("Uncasted property value should be same.", propertyValue, reference.getProperty(propertyKey));
77      assertSame("Casted property value should be same.", propertyValue, reference.getProperty(propertyKey, propertyValue.getClass()));
78    }
79  
80    @Test
81    public void should_hold_null_property_value() throws Exception {
82      final String propertyKey = "lorem";
83      final Reference<String> reference = new ReferenceImpl<String>();
84      reference.setProperty(propertyKey, null);
85      assertNull("Uncasted property value should be same.", reference.getProperty(propertyKey));
86      assertNull("Casted property value should be same.", reference.getProperty(propertyKey, String.class));
87    }
88  
89    @Test(expected = PropertyNotSetException.class)
90    public void should_fail_reading_unset_property() throws Exception {
91      final Reference<String> reference = new ReferenceImpl<String>();
92      reference.getProperty("lorem");
93    }
94  
95    @Test(expected = PropertyAlreadySetException.class)
96    public void should_fail_setting_property_which_got_already_set() throws Exception {
97      final Reference<String> reference = new ReferenceImpl<String>();
98      reference.setProperty("lorem", null);
99      reference.setProperty("lorem", null);
100   }
101 
102   @SuppressWarnings("ConstantConditions")
103   @Test(expected = NullPointerException.class)
104   public void should_fail_for_set_with_property_key_null() throws Exception {
105     final Reference<String> reference = new ReferenceImpl<String>();
106     reference.setProperty(null, "value");
107   }
108 
109   @SuppressWarnings("ConstantConditions")
110   @Test(expected = NullPointerException.class)
111   public void should_fail_for_asking_a_null_key() throws Exception {
112     final Reference<String> reference = new ReferenceImpl<String>();
113     reference.hasProperty(null);
114   }
115 
116   @Test
117   public void should_acknowledge_existing_key() {
118     final Reference<String> reference = new ReferenceImpl<String>();
119     reference.setProperty("foo", "bar");
120     assertTrue("Expected \"foo\" to be there.", reference.hasProperty("foo"));
121   }
122 
123   @Test
124   public void should_deny_existence_of_property() {
125     final Reference<String> reference = new ReferenceImpl<String>();
126     assertFalse("\"foo\" must not be set.", reference.hasProperty("foo"));
127   }
128 
129   @SuppressWarnings("ConstantConditions")
130   @Test(expected = NullPointerException.class)
131   public void should_fail_for_removing_property_with_key_null() {
132     final Reference<String> reference = new ReferenceImpl<String>();
133     reference.removeProperty(null, String.class);
134   }
135 
136   @SuppressWarnings("ConstantConditions")
137   @Test(expected = NullPointerException.class)
138   public void should_fail_for_removing_property_with_expected_class_null() {
139     final Reference<String> reference = new ReferenceImpl<String>();
140     reference.removeProperty("foo", null);
141   }
142 
143   @Test(expected = PropertyNotSetException.class)
144   public void should_fail_when_trying_to_remove_unset_property() throws Exception {
145     final Reference<String> reference = new ReferenceImpl<String>();
146     reference.removeProperty("foo", String.class);
147   }
148 
149   @Test
150   public void remove_property_should_return_former_property_value() throws Exception {
151     final Reference<String> reference = new ReferenceImpl<String>();
152     reference.setProperty("foo", "bar");
153     assertEquals("\"foo\" must be set to \"bar\".", "bar", reference.removeProperty("foo", String.class));
154   }
155 
156   @Test
157   public void remove_property_should_actually_remove_the_property() {
158     final Reference<String> reference = new ReferenceImpl<String>();
159     reference.setProperty("foo", "bar");
160     reference.removeProperty("foo", String.class);
161     assertFalse("\"foo\" must not be set anymore.", reference.hasProperty("foo"));
162   }
163 
164   @Test
165   public void should_be_possible_to_query_if_reference_value_is_set_if_set() throws Exception {
166     final Reference<String> reference = new ReferenceImpl<String>();
167     reference.set("Lorem");
168     assertTrue("Reference should signal to carry a value.", reference.hasValue());
169   }
170 
171   @Test
172   public void hasValue_should_signal_value_set_for_reference_value_null() throws Exception {
173     final Reference<String> reference = new ReferenceImpl<String>();
174     reference.set(null);
175     assertTrue("Reference should signal to carry a value.", reference.hasValue());
176   }
177 
178   @Test
179   public void should_be_possible_to_query_if_reference_value_is_set_if_unset() throws Exception {
180     final Reference<String> reference = new ReferenceImpl<String>();
181     assertFalse("Reference should signal not to carry a value.", reference.hasValue());
182   }
183 }
