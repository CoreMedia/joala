20  package net.joala.data.random;
21  
22  import javax.annotation.Nonnull;
23  
24  /**
25   * <p>
26   *   Describes random numbers to create. This interface is required by {@link AbstractRandomNumberProvider}
27   *   in order to calculate a random number in the given range.
28   * </p>
29   * @deprecated Will be removed soon.
30   */
31  @SuppressWarnings("deprecation")
32  @Deprecated
33  public interface RandomNumberType<T extends Comparable<? extends Number>> {
34    /**
35     * Type of the number.
36     * @return the type of numbers to create
37     */
38    @Nonnull
39    Class<T> getType();
40  
41    /**
42     * <p>
43     * The minimum value supported for this number type. Typically used as default minimum value.
44     * </p>
45     * @return minimum number which might be created
46     */
47    @Nonnull
48    T min();
49  
50    /**
51     * <p>
52     *   The maximum value supported for this number type. Typically used as default maximum value.
53     * </p>
54     * @return maximum number which might be created
55     */
56    @Nonnull
57    T max();
58  
59    /**
60     * <p>
61     * Create the sum of the two values.
62     * </p>
63     * @param value1 first value
64     * @param value2 second value
65     * @return sum
66     */
67    @Nonnull
68    T sum(T value1, T value2);
69  
70    /**
71     * <p>
72     *   Return the percentage of the given value. Typically used to locate a number in a given range on the
73     *   provided numerical ray.
74     * </p>
75     * @param percent percent ranging between 0 and 1
76     * @param value value to calculate the percentage of
77     * @return percentage of given value
78     */
79    @Nonnull
80    T percentOf(double percent, T value);
81  }
