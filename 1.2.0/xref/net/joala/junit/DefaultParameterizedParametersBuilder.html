20  package net.joala.junit;
21  
22  import org.junit.runner.RunWith;
23  import org.junit.runner.Runner;
24  import org.junit.runners.Parameterized;
25  
26  import java.lang.reflect.Constructor;
27  import java.util.ArrayList;
28  import java.util.Collection;
29  import java.util.Collections;
30  import java.util.List;
31  
32  import static com.google.common.base.Preconditions.checkArgument;
33  import static java.lang.String.format;
34  import static org.apache.commons.lang3.ClassUtils.isAssignable;
35  
36  /**
37   * <p>
38   * Default builder for parameters. Each added set of parameters is verified to be of the same length.
39   * In addition the provided test class is verified if it meets the specified requirements, that is
40   * has a constructor matching the provided arguments and has the correct runner configured.
41   * </p>
42   *
43   * @since 10/8/12
44   */
45  public class DefaultParameterizedParametersBuilder implements ParameterizedParametersBuilder {
46    private static final int EXPECTED_PARAMETER_SETS_COUNT = 10;
47    private final List<Object[]> parameterSets = new ArrayList<Object[]>(EXPECTED_PARAMETER_SETS_COUNT);
48    private final Class<?> testClass;
49    private final String testName;
50    private Class<?>[] parameterTypes;
51  
52    public DefaultParameterizedParametersBuilder(final Class<?> testClass) {
53      this.testClass = testClass;
54      testName = this.testClass.getName();
55    }
56  
57    @Override
58    public ParameterizedParametersBuilder add(final Object... objects) {
59      if (parameterTypes == null) {
60        validateClass();
61      }
62      final int parameterSetPosition = parameterSets.size() + 1;
63      checkArgument(
64              objects.length == parameterTypes.length,
65              format("Parameter length mismatch for set %d (expected %d, was %d)",
66                      parameterSetPosition,
67                      parameterTypes.length,
68                      objects.length));
69      for (int i = 0; i < objects.length; i++) {
70        final Object object = objects[i];
71        if (object != null) {
72          final Class<?> expectedType = parameterTypes[i];
73          checkArgument(isAssignable(object.getClass(), expectedType, true),
74                  format("Parameter type mismatch for set %d at position %d (expected %s, was %s)",
75                          parameterSetPosition,
76                          i + 1,
77                          expectedType.getName(),
78                          object.getClass().getName()
79                  ));
80        }
81      }
82      parameterSets.add(objects);
83      return this;
84    }
85  
86    private void validateClass() {
87      final RunWith runWith = testClass.getAnnotation(RunWith.class);
88      if (runWith == null) {
89        throw new ParameterizedTestMismatchException(format("Test %s has no @RunWith annotation.", testName));
90      }
91      final Class<? extends Runner> runner = runWith.value();
92      if (!Parameterized.class.isAssignableFrom(runner)) {
93        throw new ParameterizedTestMismatchException(
94                format("Test %s is not marked with expected runner %s.", testName, Parameterized.class.getName()));
95      }
96      final Constructor<?>[] constructors = testClass.getConstructors();
97      if (constructors.length == 0) {
98        throw new ParameterizedTestMismatchException(format("%s misses constructor to receive parameter values.", testName));
99      }
100     if (constructors.length > 1) {
101       throw new ParameterizedTestMismatchException(format("%s must not have more than one constructor", testName));
102     }
103     final Constructor<?> constructor = constructors[0];
104     parameterTypes = constructor.getParameterTypes();
105   }
106 
107   @Override
108   public Collection<Object[]> build() {
109     return Collections.unmodifiableList(parameterSets);
110   }
111 
112 }
