20  package net.joala.image.impl;
21  
22  import com.google.common.base.MoreObjects;
23  import net.joala.image.ImageBuilderBuildException;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import javax.activation.MimeType;
28  import javax.annotation.CheckForNull;
29  import javax.annotation.Nonnull;
30  import javax.annotation.Nullable;
31  import javax.imageio.ImageIO;
32  import javax.imageio.ImageWriter;
33  import java.util.Iterator;
34  
35  import static com.google.common.base.Optional.fromNullable;
36  import static java.lang.String.format;
37  
38  /**
39   * Abstract implementation for image builders which use ImageIO for image creation.
40   *
41   * @since 2013-02-21
42   */
43  abstract class AbstractImageIOImageBuilder extends AbstractImageBuilder {
44    private static final Logger LOG = LoggerFactory.getLogger(DefaultByteArrayImageBuilder.class);
45    /**
46     * The mime type of the image. Use default if {@code null}.
47     */
48    @CheckForNull
49    private String mimeType;
50  
51    /**
52     * Sets the mime type.
53     *
54     * @param value mime type value; {@code null} to use default value
55     */
56    protected final void setMimeType(@Nullable final MimeType value) {
57      setMimeType(value == null ? null : value.toString());
58    }
59  
60    /**
61     * Sets the mime type.
62     *
63     * @param value mime type value; {@code null} to use default value
64     */
65    protected final void setMimeType(@Nullable final String value) {
66      mimeType = value;
67    }
68  
69    /**
70     * Get mime type. Returns default if unset.
71     *
72     * @return mime type
73     */
74    @Nonnull
75    protected final String getMimeType() {
76      return fromNullable(mimeType).or(getImageBuilderConfig().getDefaultMimeType());
77    }
78  
79    /**
80     * Provide an image writer for the configured mime type. The first matching image writer
81     * will be used if there are multiple registered image writers for the given mime type.
82     *
83     * @return image writer
84     */
85    @Nonnull
86    protected final ImageWriter getImageWriter() {
87      final Iterator<ImageWriter> imageWriterIterator = ImageIO.getImageWritersByMIMEType(getMimeType());
88      if (!imageWriterIterator.hasNext()) {
89        throw new ImageBuilderBuildException(format("ImageIO: No ImageWriter available for MIME Type '%s'.", getMimeType()));
90      }
91      final ImageWriter imageWriter = imageWriterIterator.next();
92      LOG.debug("Using ImageWriter {} to create image for MIME type '{}'.", imageWriter.getClass().getName(), getMimeType());
93      return imageWriter;
94    }
95  
96    @Override
97    public String toString() {
98      return MoreObjects.toStringHelper(this)
99              .add("mimeType", mimeType)
100             .toString();
101   }
102 }
