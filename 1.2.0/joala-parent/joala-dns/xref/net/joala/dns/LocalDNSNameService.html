38  /**
39   * <p>
40   * Central class for Joala DNS. It accesses the {@link NameStore} to resolve host entries which are specified
41   * by your tests. If this fails (which is normal for all unmodified host entries) several fallback strategies
42   * are applied. For more information and how to do better in JVM 7+ please see the {@link net.joala.dns package
43   * documentation}.
44   * </p>
45   *
46   * @since 10/5/12
47   * @see net.joala.dns package documentation
48   * @deprecated Will be removed soon.
49   */
50  @SuppressWarnings("deprecation")
51  @Deprecated
52  public class LocalDNSNameService implements NameService {
53    /**
54     * A logger using System-PrintStreams.
55     */
56    private static final SystemLogger LOG = SystemLogger.getLogger(LocalDNSNameService.class);
57    /**
58     * List of fallback name services to use.
59     */
60    private static final ImmutableList<NameService> FALLBACKS;
61    /**
62     * Separator for fallback strategies.
63     */
64    private static final String FALLBACK_SEPARATOR_CHAR = ",";
65    /**
66     * Pattern to separate fallback strategy ids.
67     */
68    private static final Pattern FALLBACK_SEPARATOR = Pattern.compile(FALLBACK_SEPARATOR_CHAR);
69    /**
70     * Fallback ID for DNS Java.
71     */
72    private static final String FALLBACK_DNSJAVA_ID = "dnsjava";
73    /**
74     * Fallback ID for using JVM Default.
75     */
76    private static final String FALLBACK_DEFAULT_ID = "default";
77    /**
78     * Default fallback strategies to use.
79     */
80    private static final String FALLBACK_DEFAULTS = FALLBACK_DNSJAVA_ID + FALLBACK_SEPARATOR_CHAR + FALLBACK_DEFAULT_ID;
81    /**
82     * Property to set to control fallback strategies.
83     */
84    private static final String FALLBACKS_PROPERTY = "net.joala.dns.fallbacks";
85  
86    /**
87     * Setup fall back nameservices.
88     */
89    static {
90      final String requestedFallbacksValue = System.getProperty(FALLBACKS_PROPERTY, FALLBACK_DEFAULTS);
91      final String[] fallbacks = FALLBACK_SEPARATOR.split(requestedFallbacksValue);
92      final ImmutableList.Builder<NameService> builder = ImmutableList.builder();
93      for (final String fallback : fallbacks) {
94        addFallback(fallback, builder);
95      }
96      FALLBACKS = builder.build();
97      if (FALLBACKS.isEmpty()) {
98        LOG.info("Joala DNS Fallbacks disabled.");
99      }
100   }
101 
102   /**
103    * <p>
104    * Add fallback as requested by id. Surrounding spaces and case is ignored. Unknown fallbacks are silently
105    * ignored so that you could for example specify "disabled" in order to have no fallbacks at all.
106    * </p>
107    *
108    * @param id      the id of the fallback; supported {@code dnsjava} and {@code default}
109    * @param builder builder to add the fallback instances to
110    */
111   private static void addFallback(@Nonnull final String id, @Nonnull final ImmutableList.Builder<NameService> builder) {
112     try {
113       NameService service = null;
114       if (FALLBACK_DNSJAVA_ID.equalsIgnoreCase(id.trim())) {
115         service = new DNSJavaNameServiceDescriptor().createNameService();
116       } else if (FALLBACK_DEFAULT_ID.equalsIgnoreCase(id.trim())) {
117         service = fallbackNameService();
118       }
119       if (service != null) {
120         builder.add(service);
121         LOG.info(format("Adding Name Service %s as fallback: %s", id, service.getClass()));
122       }
123     } catch (Exception e) {
124       LOG.warn(format("Unable to create fallback service %s. Fallback ignored.", id), e);
125     }
126   }

128   @Override
129   @Nonnull
130   public InetAddress[] lookupAllHostAddr(@Nonnull final String name) throws UnknownHostException {
131     checkNotNull(name, "Hostname must not be null.");
132     final InetAddress[] fromStore = NameStore.nameStore().lookup(name);
133     if (fromStore.length > 0) {
134       return fromStore;
135     }
136     return fallbackLookupAllHostAddr(name);
137   }
138 
139   @Override
140   @Nonnull
141   public String getHostByAddr(@Nonnull final byte[] addr) throws UnknownHostException {
142     checkNotNull(addr, "Address must not be null.");
143     final Name fromStore = NameStore.nameStore().reverseLookup(InetAddress.getByAddress(addr));
144     if (fromStore != null) {
145       return fromStore.toString();
146     }
147     return fallbackGetHostByAddr(addr);
148   }
149 
150   /**
151    * <p>
152    * Go through fallbacks to resolve a given hostname.
153    * </p>
154    *
155    * @param name the hostname to resolve
156    * @return the address the hostname got resolved to
157    * @throws UnknownHostException if the address could not be resolved
158    */
159   @Nonnull
160   private InetAddress[] fallbackLookupAllHostAddr(@Nonnull final String name) throws UnknownHostException {
161     InetAddress[] fromFallbacks = null;
162     UnknownHostException firstException = null;
163     for (final NameService nameService : FALLBACKS) {
164       try {
165         fromFallbacks = nameService.lookupAllHostAddr(name);
166         break;
167       } catch (UnknownHostException e) {
168         if (firstException == null) {
169           firstException = e;
170         }
171       }
172     }
173     return handleFallbackResponse(
174             fromFallbacks,
175             firstException,
176             format("Could not find any fallback service which could resolve %s.", name));
177   }
178 
179   /**
180    * <p>
181    * Ask fallbacks for host by provided address.
182    * </p>
183    *
184    * @param addr the address to search for
185    * @return name of the host with the given address
186    * @throws UnknownHostException if such a host could not be found in the fallbacks
187    */
188   @Nonnull
189   private String fallbackGetHostByAddr(@Nonnull final byte[] addr) throws UnknownHostException {
190     String fromFallbacks = null;
191     UnknownHostException firstException = null;
192     for (final NameService nameService : FALLBACKS) {
193       try {
194         fromFallbacks = nameService.getHostByAddr(addr);
195         break;
196       } catch (UnknownHostException e) {
197         if (firstException == null) {
198           firstException = e;
199         }
200         fromFallbacks = null;
201       }
202     }
203     return handleFallbackResponse(
204             fromFallbacks,
205             firstException,
206             format("Could not find any fallback service which could resolve address %s.", Arrays.toString(addr)));
207   }
208 
209   /**
210    * <p>
211    * Handles the provided response from fallbacks.
212    * </p>
213    *
214    * @param fromFallbacks  the answer from the fallbacks; a value of {@code null} causes an {@link UnknownHostException}
215    * @param firstException the first exception provided by the fallbacks - if any; will be reused if it is required
216    *                       to throw an {@link UnknownHostException}.
217    * @param message        the message to add to exception if {@code firstException} is not used
218    * @param <T>            the response type
219    * @return either the response as provided by fallback or ends with exception
220    * @throws UnknownHostException exception if the {@code fromFallbacks} was {@code null}
221    */
222   @Nonnull
223   private <T> T handleFallbackResponse(@Nullable final T fromFallbacks,
224                                        @Nullable final UnknownHostException firstException,
225                                        @Nonnull final String message) throws UnknownHostException {
226     if (fromFallbacks == null) {
227       if (firstException == null) {
228         throw new UnknownHostException(message);
229       } else {
230         throw firstException;
231       }
232     }
233     return fromFallbacks;
234   }
