35  /**
36   * <p>
37   * Tests {@link TimeFormat}.
38   * </p>
39   *
40   * @since 9/19/12
41   */
42  @SuppressWarnings("ProhibitedExceptionDeclared")
43  @RunWith(Parameterized.class)
44  public class TimeFormatTest {
45    private final String message;
46    private final long amount;
47    private final TimeUnit timeUnit;
48    private final Pattern expectedTimePattern;
49  
50    public TimeFormatTest(final long amount, final TimeUnit timeUnit, final String expectedTimePattern) {
51      this.timeUnit = timeUnit;
52      this.expectedTimePattern = Pattern.compile(expectedTimePattern);
53      this.message = String.format("Format amount %d of %s.", amount, timeUnit);
54      this.amount = amount;
55    }
56  
57    @Test
58    public void time_should_be_formatted_as_expected() throws Exception {
59      final String description = TimeFormat.format(amount, timeUnit);
60      assertThat(message, description, new MatchesPattern(expectedTimePattern));
61    }
62  
63    @Parameterized.Parameters
64    public static Collection<Object[]> data() {
65      return defaultParametersBuilder(TimeFormatTest.class)
66              .add(TimeUnit.SECONDS.toMillis(TimeFormat.TIMEUNIT_LIMIT) - 1, TimeUnit.MILLISECONDS, "^1999\\sms$")
67              .add(TimeUnit.SECONDS.toMillis(TimeFormat.TIMEUNIT_LIMIT), TimeUnit.MILLISECONDS, "^2\\ss$")
68              .add(TimeUnit.MINUTES.toMillis(TimeFormat.TIMEUNIT_LIMIT) - 1, TimeUnit.MILLISECONDS, "^119\\ss$")
69              .add(TimeUnit.MINUTES.toMillis(TimeFormat.TIMEUNIT_LIMIT), TimeUnit.MILLISECONDS, "^2\\smin$")
70              .add(TimeUnit.HOURS.toMillis(TimeFormat.TIMEUNIT_LIMIT) - 1, TimeUnit.MILLISECONDS, "^119\\smin$")
71              .add(TimeUnit.HOURS.toMillis(TimeFormat.TIMEUNIT_LIMIT), TimeUnit.MILLISECONDS, "^2\\sh$")
72              .add(TimeUnit.DAYS.toMillis(TimeFormat.TIMEUNIT_LIMIT) - 1, TimeUnit.MILLISECONDS, "^47\\sh$")
73              .add(TimeUnit.DAYS.toMillis(TimeFormat.TIMEUNIT_LIMIT), TimeUnit.MILLISECONDS, "^2\\sd$")
74              .add(Long.MAX_VALUE, TimeUnit.MILLISECONDS, "^\\d+\\sd$")
75              .add(0L, TimeUnit.NANOSECONDS, "^0\\sns$")
76              .add(0L, TimeUnit.MICROSECONDS, "^0\\sÂµs$")
77              .add(0L, TimeUnit.MILLISECONDS, "^0\\sms$")
78              .add(0L, TimeUnit.SECONDS, "^0\\ss$")
79              .add(0L, TimeUnit.MINUTES, "^0\\smin$")
80              .add(0L, TimeUnit.HOURS, "^0\\sh$")
81              .add(0L, TimeUnit.DAYS, "^0\\sd$")
82              .build();
83    }
84  
85    // Duplicate in order to remove dependency cycles between modules.
86    private static final class MatchesPattern extends TypeSafeMatcher<CharSequence> {
87      private final Pattern pattern;
88  
89      private MatchesPattern(final Pattern pattern) {
90        this.pattern = pattern;
91      }
92  
93      @Override
94      public void describeTo(final Description description) {
95        description.appendText("a sequence of characters matching pattern ");
96        description.appendValue(pattern);
97      }
98  
99      @Override
100     protected boolean matchesSafely(final CharSequence item) {
101       return pattern.matcher(item).matches();
102     }
103   }
104 }
