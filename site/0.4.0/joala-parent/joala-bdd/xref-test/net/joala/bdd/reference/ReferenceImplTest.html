26  /**

28   * Test for {@link ReferenceImpl}.

31   * @since 6/5/12

33  public class ReferenceImplTest {
34    @Test
35    public void should_hold_value() throws Exception {
36      final Reference<Map<Object, Object>> reference = new ReferenceImpl<Map<Object, Object>>();
37      final Map<Object, Object> referenceValue = Collections.emptyMap();
38      reference.set(referenceValue);
39      assertSame("Reference value retrieved should hold the original value set.", referenceValue, reference.get());
40    }
41  
42    @Test(expected = ReferenceAlreadyBoundException.class)
43    public void should_deny_to_set_value_twice() throws Exception {
44      final Reference<String> reference = new ReferenceImpl<String>();
45      reference.set("Lorem");
46      reference.set("Ipsum");
47    }
48  
49    @Test(expected = ReferenceNotBoundException.class)
50    public void should_deny_to_read_from_unbound_reference() throws Exception {
51      final Reference<String> reference = new ReferenceImpl<String>();
52      reference.get();
53    }
54  
55    @Test(expected = NullPointerException.class)
56    public void should_fail_for_reference_value_null() throws Exception {
57      final Reference<String> reference = new ReferenceImpl<String>();
58      reference.set(null);
59    }
60  
61    @Test
62    public void should_hold_property_value() throws Exception {
63      final String propertyKey = "lorem";
64      final Map<Object, Object> propertyValue = Collections.emptyMap();
65      final Reference<String> reference = new ReferenceImpl<String>();
66      reference.setProperty(propertyKey, propertyValue);
67      assertSame("Uncasted property value should be same.", propertyValue, reference.getProperty(propertyKey));
68      assertSame("Casted property value should be same.", propertyValue, reference.getProperty(propertyKey, propertyValue.getClass()));
69    }
70  
71    @Test(expected = PropertyNotSetException.class)
72    public void should_fail_reading_unset_property() throws Exception {
73      final Reference<String> reference = new ReferenceImpl<String>();
74      reference.getProperty("lorem");
75    }
76  
77    @Test(expected = PropertyAlreadySetException.class)
78    public void should_fail_setting_property_which_got_already_set() throws Exception {
79      final Reference<String> reference = new ReferenceImpl<String>();
80      reference.setProperty("lorem", null);
81      reference.setProperty("lorem", null);
82    }
83  
84    @Test(expected = NullPointerException.class)
85    public void should_fail_for_set_with_property_key_null() throws Exception {
86      final Reference<String> reference = new ReferenceImpl<String>();

88    }
