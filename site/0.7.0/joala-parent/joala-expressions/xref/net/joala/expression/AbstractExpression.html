20  package net.joala.expression;
21  
22  import com.google.common.base.Objects;
23  import org.hamcrest.Description;
24  
25  import javax.annotation.Nonnull;
26  import javax.annotation.Nullable;
27  
28  import static com.google.common.base.Preconditions.checkNotNull;
29  
30  /**
31   * <p>
32   * Convenience abstract implementation of {@link Expression} which can get
33   * a simple description via constructor. Thus you can skip declaring the method
34   * {@link #describeTo(Description)}.
35   * </p>
36   *
37   * @param <T> the result type of the expression
38   * @since 2/27/12
39   */
40  public abstract class AbstractExpression<T> implements Expression<T> {
41    /**
42     * The simple description.
43     */
44    @Nullable
45    private final String simpleDescription;
46  
47    protected AbstractExpression() {
48      this(null);
49    }
50  
51    /**
52     * <p>
53     * Constructor with a simple description.
54     * </p>
55     *
56     * @param simpleDescription description to add; {@code null} if this object shall not provide any description
57     */
58    protected AbstractExpression(@Nullable final String simpleDescription) {
59      this.simpleDescription = simpleDescription;
60    }
61  
62    /**
63     * <p>
64     * Will add the provided simple description to the provided description.
65     * </p>
66     *
67     * @param description The description to be built or appended to.
68     */
69    @Override
70    public void describeTo(@Nonnull final Description description) {
71      checkNotNull(description, "Description must not be null.");
72      if (simpleDescription != null) {
73        description.appendText(simpleDescription);
74      }
75    }
76  
77    /**
78     * <p>
79     * String representation for debugging purpose only.
80     * </p>
81     *
82     * @return string representation
83     */
84    @Override
85    public String toString() {
86      return Objects.toStringHelper(this)
87              .add("simpleDescription", simpleDescription)
88              .toString();
89    }
90  }
