48  /**
49   * <p>
50   * Tests some requirements for the {@code toString()} method. See the test methods
51   * for the tested requirements.
52   * </p>

56  @Ignore("Don't execute Testlets in IDE and alike.")
57  @SuppressWarnings("JUnitTestClassNamingConvention")
58  public class ToStringTestlet<T> extends AbstractTestlet<T> {
59    /**
60     * Logging instance.
61     */
62    private static final Logger LOG = LoggerFactory.getLogger(ToStringTestlet.class);

65     * Fields collected from the given class.

67    @Nonnull
68    private Class<? super T> fieldsFromClass;
69    private boolean constantFields;

72    private boolean enumFields = true;

79     * Constructor with the testling.

84    @SuppressWarnings("unchecked")

86      super(testling);

90    @Test
91    public void toString_should_contain_classname() {
92      if (!ignoreClassname) {
93        assertThat("toString should contain classname", getTestling().toString(), containsString(getTestling().getClass().getSimpleName()));
94      }
95    }
96  
97    @SuppressWarnings("unchecked")
98    @Test
99    public void toString_should_contain_all_fields_and_their_values() throws IllegalAccessException {
100     final Field[] declaredFields = getExpectedFields();
101     LOG.debug("Validating {} fields of {}.", declaredFields.length, fieldsFromClass);
102     final Collection<Matcher<? super String>> fieldMatchers = new ArrayList<Matcher<? super String>>(declaredFields.length * 2);
103     for (final Field declaredField : declaredFields) {
104       if (!Modifier.isStatic(declaredField.getModifiers())) {
105         doPrivileged(new SetAccessibleAction(declaredField));
106         final String fieldName = declaredField.getName();
107         fieldMatchers.add(containsString(fieldName));
108         fieldMatchers.add(containsString(String.valueOf(declaredField.get(getTestling()))));
109         LOG.debug("Added validator for field {} of class {}.", fieldName, fieldsFromClass);
110       }
111     }
112     assertThat("toString should contain all fields and their values", getTestling().toString(), allOf(fieldMatchers));
113   }
114 
115   @Nonnull
116   private Field[] getExpectedFields() {
117     final List<Field> fields = Arrays.asList(fieldsFromClass.getDeclaredFields());
118     final Collection<Field> filteredFields = Collections2.filter(fields, new FieldPredicate());
119     return filteredFields.toArray(new Field[filteredFields.size()]);
120   }

234   private class FieldPredicate implements Predicate<Field> {
235     @SuppressWarnings("RedundantIfStatement")
236     @Override
237     public boolean apply(final Field input) {
238       final int modifiers = input.getModifiers();
239       if (!constantFields && Modifier.isStatic(modifiers) && Modifier.isFinal(modifiers)) {
240         return false;
241       }
242       if (!staticFields && Modifier.isStatic(modifiers)) {
243         return false;
244       }
245       if (!transientFields && Modifier.isTransient(modifiers)) {
246         return false;
247       }
248       if (!enumFields && input.isEnumConstant()) {
249         return false;
250       }
251       if (excludeNamePattern != null && excludeNamePattern.matcher(input.getName()).matches()) {
252         return false;
253       }
254       if (excludeTypeNamePattern != null && excludeTypeNamePattern.matcher(input.getType().getName()).matches()) {
255         return false;
256       }
257       return true;
258     }
