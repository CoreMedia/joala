52    @Test
53    public void toString_should_contain_all_fields_and_their_values() throws IllegalAccessException {
54      final Field[] declaredFields = fieldDeclaringClass.getDeclaredFields();
55      LOG.debug("Validating {} fields of {}.", declaredFields.length, fieldDeclaringClass);
56      final Collection<Matcher<? super String>> fieldMatchers = new ArrayList<Matcher<? super String>>(declaredFields.length * 2);
57      for (final Field declaredField : declaredFields) {
58        if (!Modifier.isStatic(declaredField.getModifiers())) {
59          declaredField.setAccessible(true);
60          final String fieldName = declaredField.getName();
61          fieldMatchers.add(containsString(fieldName));
62          fieldMatchers.add(containsString(String.valueOf(declaredField.get(objectUnderTest))));
63          LOG.debug("Added validator for field {} of class {}.", fieldName, fieldDeclaringClass);
64        }
65      }
66      assertThat("toString should contain all fields and their values", objectUnderTest.toString(), allOf(fieldMatchers));
67    }
