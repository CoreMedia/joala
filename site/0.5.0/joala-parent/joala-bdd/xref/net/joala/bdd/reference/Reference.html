6   /**
7    * <p>

13   * </p>

17   */

21     * Set reference value. Can only be called once.
22     * </p>
23     *
24     * @param value the value of the reference
25     * @throws ReferenceAlreadyBoundException if the reference already has a value
26     * @throws NullPointerException           if the value set is null
27     */
28    void set(@Nonnull T value);

31     * <p>
32     * Retrieve the value of a reference.
33     * </p>
34     *
35     * @return the value of the reference
36     * @throws ReferenceNotBoundException if the reference does not contain a value yet
37     */
38    @Nonnull
39    T get();
40  
41    /**
42     * <p>
43     * Set a property which belongs to the reference value. Such a property might for example be a text which will
44     * be represent the reference value in a webapp. For example a username or something similar.
45     * </p>
46     *
47     * @param key   the name of the property
48     * @param value the value of the property
49     * @throws NullPointerException        if the key is {@code null}

63    @Nullable
