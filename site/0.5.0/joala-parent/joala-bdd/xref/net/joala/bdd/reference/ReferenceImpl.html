41    }

50    }

52    @Override
53    public void setProperty(@Nonnull final String key, @Nullable final Object value) {
54      checkNotNull(key, "Property key must not be null.");
55      if (properties.containsKey(key)) {
56        throw new PropertyAlreadySetException(format("Property '%s' already set to value %s", key, properties.get(key)));
57      }
58      properties.put(key, value);
59    }
60  
61    @Override
62    public Object getProperty(@Nonnull final String key) {
63      checkNotNull(key, "Property key must not be null.");
64      if (!properties.containsKey(key)) {
65        throw new PropertyNotSetException(format("Property '%s' not set.", key));
66      }
67      return properties.get(key);
68    }
69  
70    @Override
71    @Nullable
72    public <P> P getProperty(@Nonnull final String key, @Nonnull final Class<P> clazz) {
73      checkNotNull(key, "Property key must not be null.");
74      final Object propertyValue = getProperty(key);
75      if (propertyValue == null) {

77      }

79    }
