39    @Nonnull
40    private final PreparedResponsesHttpHandler preparedResponsesHttpHandler;

44    @Nonnull

64      this("/");

67    /**
68     * <p>
69     * Creates a webservice responding to a random free port on specified context.
70     * </p>
71     *
72     * @param context context, for example {@code /lorem} for lorem-context
73     * @throws IOException on failure
74     */
75    public EmbeddedWebservice(@Nonnull final String context) throws IOException {
76      this(context, freePort());
77    }
78  
79    /**
80     * <p>
81     * Creates a webservice responding to given port on specified context.
82     * </p>
83     *
84     * @param context context, for example {@code /lorem} for lorem-context
85     * @param port    port to bind the service to
86     * @throws IOException on failure
87     */
88    public EmbeddedWebservice(@Nonnull final String context, final int port) throws IOException {
89      checkNotNull(context, "Context must not be null. For root context use '/'.");
90      this.context = context;
91      this.port = port;
92      final InetSocketAddress address = new InetSocketAddress(port);
93      server = HttpServer.create(address, 0);
94      preparedResponsesHttpHandler = new PreparedResponsesHttpHandlerImpl();
95      server.createContext(this.context, preparedResponsesHttpHandler);
96      server.setExecutor(null);
97      clientUri = URI.create(String.format("http://%s:%d%s", getByName(null).getHostName(), this.port, context));
98    }
99

108   public URI getClientUri() {
109     return clientUri;

115   public void start() {
116     checkState(server != null, "Server cannot be restarted.");
117     server.start();
118     LOG.debug("Started embedded webservice at port {} with context {}.", port, context);
119   }

121   /**
122    * Stop the webservice.
123    */
124   public void stop() {
125     server.stop(1);
126     preparedResponsesHttpHandler.clearResponses();
127     server = null;

129   }
