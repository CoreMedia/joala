20  package net.joala.matcher.reflect;
21  
22  import org.hamcrest.Description;
23  import org.hamcrest.Factory;
24  import org.hamcrest.Matcher;
25  import org.hamcrest.TypeSafeMatcher;
26  
27  import java.lang.reflect.Field;
28  import java.lang.reflect.Member;
29  import java.lang.reflect.Method;
30  import java.lang.reflect.Modifier;
31  
32  /**
33   * @since 10/9/12
34   */
35  public class MemberHasModifier<T extends Member> extends TypeSafeMatcher<T> {
36    private final int modifierFlag;
37  
38    public MemberHasModifier(final int modifierFlag) {
39      this.modifierFlag = modifierFlag;
40    }
41  
42    @Override
43    protected boolean matchesSafely(final T item) {
44      return (item.getModifiers() & modifierFlag) != 0;
45    }
46  
47    @Override
48    public void describeTo(final Description description) {
49      description
50              .appendText("is ")
51              .appendText(Modifier.toString(modifierFlag));
52    }
53  
54    @Override
55    protected void describeMismatchSafely(final T item, final Description description) {
56      description
57              .appendText("but was ")
58              .appendText(Modifier.toString(item.getModifiers()));
59    }
60  
61    @Factory
62    public static <T extends Method> Matcher<T> memberIsAbstract() {
63      return new MemberHasModifier<T>(Modifier.ABSTRACT);
64    }
65  
66    @Factory
67    public static <T extends Member> Matcher<T> memberIsFinal() {
68      return new MemberHasModifier<T>(Modifier.FINAL);
69    }
70  
71    @Factory
72    public static <T extends Method> Matcher<T> memberIsNative() {
73      return new MemberHasModifier<T>(Modifier.NATIVE);
74    }
75  
76    @Factory
77    public static <T extends Member> Matcher<T> memberIsPrivate() {
78      return new MemberHasModifier<T>(Modifier.PRIVATE);
79    }
80  
81    @Factory
82    public static <T extends Member> Matcher<T> memberIsProtected() {
83      return new MemberHasModifier<T>(Modifier.PROTECTED);
84    }
85  
86    @Factory
87    public static <T extends Member> Matcher<T> memberIsPublic() {
88      return new MemberHasModifier<T>(Modifier.PUBLIC);
89    }
90  
91    @Factory
92    public static <T extends Member> Matcher<T> memberIsStatic() {
93      return new MemberHasModifier<T>(Modifier.STATIC);
94    }
95  
96    @Factory
97    public static <T extends Method> Matcher<T> memberIsStrict() {
98      return new MemberHasModifier<T>(Modifier.STRICT);
99    }
100 
101   @Factory
102   public static <T extends Method> Matcher<T> memberIsSynchronized() {
103     return new MemberHasModifier<T>(Modifier.SYNCHRONIZED);
104   }
105 
106   @Factory
107   public static <T extends Field> Matcher<T> memberIsVolatile() {
108     return new MemberHasModifier<T>(Modifier.VOLATILE);
109   }
