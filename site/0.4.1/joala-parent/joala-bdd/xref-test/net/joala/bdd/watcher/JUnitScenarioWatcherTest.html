169   private void headersShouldBeAligned(final Appendable log, final Class<?> testClass) {
170     runClasses(testClass);
171     final String logged = log.toString().toLowerCase();
172     final String[] strings = NEWLINE_PATTERN.split(logged);
173     assertEquals("Three lines should be contained in the report.", 3, strings.length);
174     String referenceLine = null;
175     for (final String string : strings) {
176       final Matcher matcher = LOGGED_MESSAGE_PATTERN.matcher(string);
177       assumeTrue(matcher.find());
178       if (referenceLine == null) {
179         referenceLine = matcher.group(1);
180       } else {
181         final String currentLine = matcher.group(1);
182         assertEquals(String.format("Padding for all lines should be equal (reference: %s, current: %s)", referenceLine, currentLine), referenceLine.length(), currentLine.length());
183       }
184     }
185   }

187   private void scenarioStoryHeadersShouldBeReported(final Appendable log, final Class<?> testClass) {
188     runClasses(testClass);
189     final String logged = log.toString().toLowerCase();
190     assertThat("Story header should be logged.", logged, containsString(STORY_HEADING));
191     assertThat("Scenario header should be logged.", logged, containsString(SCENARIO_HEADING));
192   }

194   private static final class MockJUnitScenarioWatcher extends JUnitScenarioWatcher {
195     private final Appendable appendable;
196 
197     private MockJUnitScenarioWatcher(final Appendable appendable) {
198       this.appendable = appendable;
199     }
200 
201     @Override
202     protected void report(final String message) {
203       try {
204         appendable.append(message);
205         appendable.append(System.getProperty("line.separator", "\n"));
206       } catch (IOException e) {
207         throw new RuntimeException(e);
208       }
209     }
210   }
