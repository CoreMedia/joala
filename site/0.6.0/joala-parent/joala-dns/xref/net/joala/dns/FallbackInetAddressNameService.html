13  /**
14   * <p>
15   * Fallback DNS resolution strategy using Java Default Implementation. As noted in the
16   * {@link net.joala.dns package description} this is actually a hack using Java Reflection
17   * to access private methods and fields. Thus it will most likely fail with some JVM versions.
18   * Since Java 7 it is recommended to provide your own order of DNS providers using an ordered
19   * list of {@code sun.net.spi.nameservice.provider.#}.
20   * </p>
21   *
22   * @since 10/6/12
23   * @see net.joala.dns package documentation
24   */
25  final class FallbackInetAddressNameService implements NameService {

27     * A logger using System-PrintStreams.
28     */
29    private static final SystemLogger LOG = SystemLogger.getLogger(FallbackInetAddressNameService.class);
30    /**
31     * Class (verified to exist in Java 6 & 7) that is responsible for creating instances of
32     * {@code InetAddressImpl}.
33     */
34    private static final String JAVA_NET_INET_ADDRESS_IMPL_FACTORY = "java.net.InetAddressImplFactory";
35    /**
36     * Method hidden in {@code InetAddressImpl}.
37     */
38    private static final String LOOKUP_ALL_HOST_ADDR = "lookupAllHostAddr";
39    /**
40     * Method hidden in {@code InetAddressImpl}.
41     */
42    private static final String GET_HOST_BY_ADDR = "getHostByAddr";

44    /**
45     * Threadsafe lazy initialization pattern.
46     */
47    @SuppressWarnings("UtilityClassWithoutPrivateConstructor")
48    private static class NameServiceHolder {
49      private static final FallbackInetAddressNameService ourInstance = new FallbackInetAddressNameService();
50    }

54     * Retrieve singleton instance. On failure it is expected that either the JVM does not provide the classes/methods
55     * accessed in here or that the installed SecurityManager prohibits any access.
56     * </p>
57     *
58     * @return instance of the fallback service
59     */
60    public static FallbackInetAddressNameService fallbackNameService() {
61      return NameServiceHolder.ourInstance;

65     * Reference to the wrapped {@code InetAddressImpl}.
66     */
67    private final Object inetAddressImpl;
68    /**
69     * Method in {@code InetAddressImpl} to lookup all hosts for a given hostname.
70     */
71    private final Method lookupAllHostAddrMethod;
72    /**
73     * Method in {@code InetAddressImpl} to lookup a host by its address.
74     */
75    private final Method getHostByAddrMethod;

77    /**
78     * <p>
79     * Constructor. On failure it is expected that either the JVM does not provide the classes/methods
80     * accessed in here or that the installed SecurityManager prohibits any access.
81     * </p>
82     */
83    private FallbackInetAddressNameService() {
84      try {
85        inetAddressImpl = getInetAddressImpl();
86        lookupAllHostAddrMethod = getLookupAllHostAddrMethod();
87        getHostByAddrMethod = getGetHostByAddrMethod();
88      } catch (Exception e) {
89        throw new RuntimeException("Unable to access JVM private API as it seems.", e);
90      }
91    }

93    /**
94     * Retrieve accessible method.
95     *
96     * @return method, which is accessible
97     * @throws NoSuchMethodException if the method does not exist; unsupported JVM is assumed
98     */
99    private Method getGetHostByAddrMethod() throws NoSuchMethodException {
100     final Class<?> wrappedClass = inetAddressImpl.getClass();
101     final Method method = wrappedClass.getMethod(GET_HOST_BY_ADDR, byte[].class);
102     method.setAccessible(true);
103     return method;
104   }

107    * <p>
108    * Retrieve accessible method.
109    * </p>
110    *
111    * @return method, which is accessible
112    * @throws NoSuchMethodException if the method does not exist; unsupported JVM is assumed
113    */
114   private Method getLookupAllHostAddrMethod() throws NoSuchMethodException {
115     final Class<?> wrappedClass = inetAddressImpl.getClass();
116     final Method method = wrappedClass.getMethod(LOOKUP_ALL_HOST_ADDR, String.class);
117     method.setAccessible(true);
118     return method;
119   }

121   /**
122    * <p>
123    * Retrieve instance of {@code InetAddressImpl}.
124    * </p>
125    *
126    * @return instance
127    * @throws NoSuchMethodException     if the factory for {@code InetAddressImpl} does not provide the expected method;
128    *                                   unsupported JVM is assumed
129    * @throws ClassNotFoundException    if the factory for {@code InetAddressImpl}  is unknown; unsupported JVM is assumed
130    * @throws IllegalAccessException    if for example a security manager prohibits access
131    * @throws InvocationTargetException if it failed to call the factory method
132    */
133   private Object getInetAddressImpl() throws ClassNotFoundException, NoSuchMethodException, IllegalAccessException, InvocationTargetException {
134     final Class<?> factory = Class.forName(JAVA_NET_INET_ADDRESS_IMPL_FACTORY, true, InetAddress.class.getClassLoader());
135     final Method method = factory.getDeclaredMethod("create");
136     method.setAccessible(true);
137     return method.invoke(null);
138   }

140   @Override
141   public InetAddress[] lookupAllHostAddr(final String host) throws UnknownHostException {
142     try {
143       LOG.info("Redirecting lookupAllHostAddr() to " + inetAddressImpl.getClass().getName());
144       return (InetAddress[]) lookupAllHostAddrMethod.invoke(inetAddressImpl, host);
145     } catch (IllegalAccessException e) {
146       throw createForwardException(LOOKUP_ALL_HOST_ADDR, e);
147     } catch (InvocationTargetException e) {
148       throw createForwardException(LOOKUP_ALL_HOST_ADDR, e);
149     }
150   }
151 
152   @Override
153   public String getHostByAddr(final byte[] addr) throws UnknownHostException {
154     try {
155       LOG.info("Redirecting getHostByAddr() to " + inetAddressImpl.getClass().getName());
156       return (String) getHostByAddrMethod.invoke(inetAddressImpl, new Object[]{addr});
157     } catch (IllegalAccessException e) {
158       throw createForwardException(GET_HOST_BY_ADDR, e);
159     } catch (InvocationTargetException e) {
160       throw createForwardException(GET_HOST_BY_ADDR, e);
161     }
162   }

164   /**
165    * <p>
166    * Adds especially a cause to the {@link UnknownHostException}.
167    * </p>
168    *
169    * @param methodName method which failed
170    * @param cause      cause
171    * @return exception with a cause
172    */
173   private UnknownHostException createForwardException(final String methodName, final Throwable cause) {
174     final UnknownHostException exception = new UnknownHostException(format("Unable to forward request to %s", methodName));
175     exception.initCause(cause);
176     return exception;
177   }

179   @Override
180   public String toString() {

184             .add("lookupAllHostAddrMethod", lookupAllHostAddrMethod)
