40   * <p>
41   * Base test for classes implementing {@link WaitFailStrategy}.

46  @RunWith(MockitoJUnitRunner.class)
47  public abstract class WaitFailStrategyTest<S extends WaitFailStrategy, T extends Throwable> {
48    private static final DataProvider<String> FAIL_MESSAGE_PROVIDER = new DefaultRandomStringProvider().prefix("failMessage").fixate();
49    private static final DataProvider<String> EXCEPTION_MESSAGE_PROVIDER = new DefaultRandomStringProvider().prefix("exceptionMessage").fixate();
50    private static final DataProvider<Long> CONSUMED_MILLIS_PROVIDER = new RandomLongProvider().min(0L).max(1000L).fixate();
51  
52    @SuppressWarnings("UnusedDeclaration")
53    @Mock
54    private Object failedFunction;
55    @SuppressWarnings("UnusedDeclaration")
56    @Mock
57    private Object failedInput;
58    @SuppressWarnings("UnusedDeclaration")
59    @Mock
60    private Object failedLastValue;
61    @SuppressWarnings("UnusedDeclaration")
62    @Mock
63    private Throwable lastFailure;
64  
65    @Test
66    public void failed_match_method_should_throw_exception() throws Exception {
67      final S strategy = getFailStrategy();
68      boolean success = false;
69      try {
70        strategy.fail(FAIL_MESSAGE_PROVIDER.get(), failedFunction, failedInput, failedLastValue, nullValue(), CONSUMED_MILLIS_PROVIDER.get());
71        success = true;
72      } catch (RuntimeException ignored) {
73        // fine, that's what we expected
74      } catch (Error ignored) {
75        // fine that's what we expected
76      } catch (Throwable t) {
77        fail(format("Unexpected exception of type %s thrown.", t.getClass()));
78      }
79      assertFalse("An exception should have been thrown.", success);
80    }
81  
82    @Test
83    public void failed_with_exceptions_method_should_throw_exception() throws Exception {
84      final S strategy = getFailStrategy();
85      boolean success = false;
86      try {
87        strategy.fail(FAIL_MESSAGE_PROVIDER.get(), failedFunction, failedInput, lastFailure, CONSUMED_MILLIS_PROVIDER.get());
88        success = true;
89      } catch (RuntimeException ignored) {
90        // fine, that's what we expected
91      } catch (Error ignored) {
92        // fine that's what we expected
93      } catch (Throwable t) {
94        fail(format("Unexpected exception of type %s thrown.", t.getClass()));
95      }
96      assertFalse("An exception should have been thrown.", success);
97    }

99    @Test
100   public void message_on_failed_match_should_be_contained_in_exception() throws Throwable {
101     final S strategy = getFailStrategy();
102     final String message = FAIL_MESSAGE_PROVIDER.get();
103     boolean success = false;
104     try {
105       strategy.fail(message, failedFunction, failedInput, failedLastValue, nullValue(), CONSUMED_MILLIS_PROVIDER.get());
106       success = true;
107     } catch (Throwable t) {
108       if (!getRaisedExceptionType().isAssignableFrom(t.getClass())) {
109         throw t;
110       }
111       assertThat("Message should be contained in exception message.", t.getMessage(), containsString(message));
112     }
113     assertFalse("An exception should have been thrown.", success);
114   }

133   @Test
134   public void cause_when_failed_with_exception_should_be_contained_in_exception() throws Throwable {
135     final S strategy = getFailStrategy();
136     final String evaluationExceptionMessage = EXCEPTION_MESSAGE_PROVIDER.get();
137     when(lastFailure.getMessage()).thenReturn(evaluationExceptionMessage);
138     boolean success = false;
139     try {
140       strategy.fail(FAIL_MESSAGE_PROVIDER.get(), failedFunction, failedInput, lastFailure, CONSUMED_MILLIS_PROVIDER.get());
141       success = true;
142     } catch (Throwable t) {
143       if (!getRaisedExceptionType().isAssignableFrom(t.getClass())) {
144         throw t;
145       }
146       assertThat("Cause should be mentioned in failure.", t,
147               anyOf(
148                       causedBy(lastFailure),
149                       messageContains(lastFailure.getMessage(), true)
150               )
151       );
152     }
153     assertFalse("An exception should have been thrown.", success);
154   }
