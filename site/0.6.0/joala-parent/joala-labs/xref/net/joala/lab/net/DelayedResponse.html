36    public DelayedResponse(final long delayMillis, @Nonnull final Response response) {

39      this.response = response;
40    }
41  
42    @Override
43    public void write(@Nonnull final HttpExchange exchange) throws IOException {
44      checkNotNull(exchange, "Exchange must not be null.");
45      try {
46        Thread.sleep(delayMillis);
47      } catch (InterruptedException ignored) {
48      }
49      response.write(exchange);
50    }
51  
52    /**

56     * @param response    the response to send after delay

59    public static Response delay(final long delayMillis, @Nonnull final Response response) {
