11   * <p>
12   * Disable caching for inet addresses by reflection.
13   * </p>
14   *
15   * @since 10/7/12
16   */
17  class ReflectionInetCacheDisabler {
18    /**
19     * Class to access by reflection.
20     */
21    private static final String SUN_NET_INET_ADDRESS_CACHE_POLICY = "sun.net.InetAddressCachePolicy";
22    /**
23     * Member variable of cache policy with the policy for positive lookups.
24     */
25    private static final String CACHE_POLICY = "cachePolicy";
26    /**
27     * Member variable of cache policy with policy for negative lookups.
28     */
29    private static final String NEGATIVE_CACHE_POLICY = "negativeCachePolicy";
30    /**
31     * Signals to never cache any entries.
32     */
33    private static final int CACHE_NEVER = 0;
34  
35    /**
36     * <p>
37     * Try to disable caching by reflection.
38     * </p>
39     *
40     * @throws ReflectionCallException if disabling cache failed
41     */
42    @SuppressWarnings("unchecked")
43    public void disable() throws ReflectionCallException {
44      try {
45        final Class<?> policyClass = Class.forName(SUN_NET_INET_ADDRESS_CACHE_POLICY, true, InetAddress.class.getClassLoader());
46        final Field cachePolicyField = policyClass.getDeclaredField(CACHE_POLICY);
47        final Field negativeCachePolicyField = policyClass.getDeclaredField(NEGATIVE_CACHE_POLICY);
48        doPrivileged(new SetAccessibleAction(cachePolicyField));
49        doPrivileged(new SetAccessibleAction(negativeCachePolicyField));
50        cachePolicyField.set(null, CACHE_NEVER);
51        negativeCachePolicyField.set(null, CACHE_NEVER);
52      } catch (Exception e) { // NOSONAR: We want to be as robust as possible; thus catching all exceptions
53        throw new ReflectionCallException(e);
54      }
55    }
56  }
