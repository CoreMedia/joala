19  /**
20   * <p>
21   * Central class for Joala DNS. It accesses the {@link NameStore} to resolve host entries which are specified
22   * by your tests. If this fails (which is normal for all unmodified host entries) several fallback strategies
23   * are applied. For more information and how to do better in JVM 7+ please see the {@link net.joala.dns package
24   * documentation}.
25   * </p>
26   *
27   * @since 10/5/12
28   * @see net.joala.dns package documentation
29   */
30  public class LocalDNSNameService implements NameService {
31    /**
32     * A logger using System-PrintStreams.
33     */
34    private static final SystemLogger LOG = SystemLogger.getLogger(LocalDNSNameService.class);
35    /**
36     * List of fallback name services to use.
37     */
38    private static final ImmutableList<NameService> FALLBACKS;
39    /**
40     * Separator for fallback strategies.
41     */
42    private static final String FALLBACK_SEPARATOR_CHAR = ",";
43    /**
44     * Pattern to separate fallback strategy ids.
45     */
46    private static final Pattern FALLBACK_SEPARATOR = Pattern.compile(FALLBACK_SEPARATOR_CHAR);
47    /**
48     * Fallback ID for DNS Java.
49     */
50    private static final String FALLBACK_DNSJAVA_ID = "dnsjava";
51    /**
52     * Fallback ID for using JVM Default.
53     */
54    private static final String FALLBACK_DEFAULT_ID = "default";
55    /**
56     * Default fallback strategies to use.
57     */
58    private static final String FALLBACK_DEFAULTS = FALLBACK_DNSJAVA_ID + FALLBACK_SEPARATOR_CHAR + FALLBACK_DEFAULT_ID;
59    /**
60     * Property to set to control fallback strategies.
61     */
62    private static final String FALLBACKS_PROPERTY = "net.joala.dns.fallbacks";
63  
64    /**
65     * Setup fall back nameservices.
66     */
67    static {
68      final String requestedFallbacksValue = System.getProperty(FALLBACKS_PROPERTY, FALLBACK_DEFAULTS);
69      final String[] fallbacks = FALLBACK_SEPARATOR.split(requestedFallbacksValue);
70      final ImmutableList.Builder<NameService> builder = ImmutableList.builder();
71      for (final String fallback : fallbacks) {
72        addFallback(fallback, builder);
73      }
74      FALLBACKS = builder.build();
75      if (FALLBACKS.isEmpty()) {
76        LOG.info("Joala DNS Fallbacks disabled.");
77      }
78    }
79  
80    /**
81     * <p>
82     * Add fallback as requested by id. Surrounding spaces and case is ignored. Unknown fallbacks are silently
83     * ignored so that you could for example specify "disabled" in order to have no fallbacks at all.
84     * </p>
85     *
86     * @param id      the id of the fallback; supported {@code dnsjava} and {@code default}
87     * @param builder builder to add the fallback instances to
88     */
89    private static void addFallback(@Nonnull final String id, @Nonnull final ImmutableList.Builder<NameService> builder) {
90      try {
91        NameService service = null;
92        if (FALLBACK_DNSJAVA_ID.equalsIgnoreCase(id.trim())) {
93          service = new DNSJavaNameServiceDescriptor().createNameService();
94        } else if (FALLBACK_DEFAULT_ID.equalsIgnoreCase(id.trim())) {
95          service = fallbackNameService();
96        }
97        if (service != null) {
98          builder.add(service);
99          LOG.info(format("Adding Name Service %s as fallback: %s", id, service.getClass()));
100       }
101     } catch (Exception e) {
102       LOG.warn(format("Unable to create fallback service %s. Fallback ignored.", id), e);
103     }
104   }

106   @Override
107   @Nonnull
108   public InetAddress[] lookupAllHostAddr(@Nonnull final String name) throws UnknownHostException {
109     checkNotNull(name, "Hostname must not be null.");
110     final InetAddress[] fromStore = NameStore.nameStore().lookup(name);
111     if (fromStore.length > 0) {
112       return fromStore;
113     }
114     return fallbackLookupAllHostAddr(name);
115   }

117   @Override
118   @Nonnull
119   public String getHostByAddr(@Nonnull final byte[] addr) throws UnknownHostException {
120     checkNotNull(addr, "Address must not be null.");
121     final Name fromStore = NameStore.nameStore().reverseLookup(InetAddress.getByAddress(addr));
122     if (fromStore != null) {
123       return fromStore.toString();
124     }
125     return fallbackGetHostByAddr(addr);
126   }

128   /**
129    * <p>
130    * Go through fallbacks to resolve a given hostname.
131    * </p>
132    *
133    * @param name the hostname to resolve
134    * @return the address the hostname got resolved to
135    * @throws UnknownHostException if the address could not be resolved
136    */
137   @Nonnull
138   private InetAddress[] fallbackLookupAllHostAddr(@Nonnull final String name) throws UnknownHostException {
139     InetAddress[] fromFallbacks = null;
140     UnknownHostException firstException = null;
141     for (final NameService nameService : FALLBACKS) {
142       try {
143         fromFallbacks = nameService.lookupAllHostAddr(name);
144         break;
145       } catch (UnknownHostException e) {
146         if (firstException == null) {
147           firstException = e;
148         }
149       }
150     }
151     return handleFallbackResponse(
152             fromFallbacks,
153             firstException,
154             format("Could not find any fallback service which could resolve %s.", name));
155   }

157   /**
158    * <p>
159    * Ask fallbacks for host by provided address.
160    * </p>
161    *
162    * @param addr the address to search for
163    * @return name of the host with the given address
164    * @throws UnknownHostException if such a host could not be found in the fallbacks
165    */
166   @Nonnull
167   private String fallbackGetHostByAddr(@Nonnull final byte[] addr) throws UnknownHostException {
168     String fromFallbacks = null;
169     UnknownHostException firstException = null;
170     for (final NameService nameService : FALLBACKS) {
171       try {
172         fromFallbacks = nameService.getHostByAddr(addr);
173         break;
174       } catch (UnknownHostException e) {
175         if (firstException == null) {
176           firstException = e;
177         }
178         fromFallbacks = null;
179       }
180     }
181     return handleFallbackResponse(
182             fromFallbacks,
183             firstException,
184             format("Could not find any fallback service which could resolve address %s.", Arrays.toString(addr)));
185   }

187   /**
188    * <p>
189    * Handles the provided response from fallbacks.
190    * </p>
191    *
192    * @param fromFallbacks  the answer from the fallbacks; a value of {@code null} causes an {@link UnknownHostException}
193    * @param firstException the first exception provided by the fallbacks - if any; will be reused if it is required
194    *                       to throw an {@link UnknownHostException}.
195    * @param message        the message to add to exception if {@code firstException} is not used
196    * @param <T>            the response type
197    * @return either the response as provided by fallback or ends with exception
198    * @throws UnknownHostException exception if the {@code fromFallbacks} was {@code null}
199    */
200   @Nonnull
201   private <T> T handleFallbackResponse(@Nullable final T fromFallbacks,
202                                        @Nullable final UnknownHostException firstException,
203                                        @Nonnull final String message) throws UnknownHostException {
204     if (fromFallbacks == null) {
205       if (firstException == null) {
206         throw new UnknownHostException(message);
207       } else {
208         throw firstException;
209       }
210     }
211     return fromFallbacks;
212   }
