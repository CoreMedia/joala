38  /**
39   * <p>
40   * Central class for Joala DNS. It accesses the {@link NameStore} to resolve host entries which are specified
41   * by your tests. If this fails (which is normal for all unmodified host entries) several fallback strategies
42   * are applied. For more information and how to do better in JVM 7+ please see the {@link net.joala.dns package
43   * documentation}.
44   * </p>
45   *
46   * @since 10/5/12
47   * @see net.joala.dns package documentation
48   */
49  public class LocalDNSNameService implements NameService {
50    /**
51     * A logger using System-PrintStreams.
52     */
53    private static final SystemLogger LOG = SystemLogger.getLogger(LocalDNSNameService.class);
54    /**
55     * List of fallback name services to use.
56     */
57    private static final ImmutableList<NameService> FALLBACKS;
58    /**
59     * Separator for fallback strategies.
60     */
61    private static final String FALLBACK_SEPARATOR_CHAR = ",";
62    /**
63     * Pattern to separate fallback strategy ids.
64     */
65    private static final Pattern FALLBACK_SEPARATOR = Pattern.compile(FALLBACK_SEPARATOR_CHAR);
66    /**
67     * Fallback ID for DNS Java.
68     */
69    private static final String FALLBACK_DNSJAVA_ID = "dnsjava";
70    /**
71     * Fallback ID for using JVM Default.
72     */
73    private static final String FALLBACK_DEFAULT_ID = "default";
74    /**
75     * Default fallback strategies to use.
76     */
77    private static final String FALLBACK_DEFAULTS = FALLBACK_DNSJAVA_ID + FALLBACK_SEPARATOR_CHAR + FALLBACK_DEFAULT_ID;
78    /**
79     * Property to set to control fallback strategies.
80     */
81    private static final String FALLBACKS_PROPERTY = "net.joala.dns.fallbacks";
82  
83    /**
84     * Setup fall back nameservices.
85     */
86    static {
87      final String requestedFallbacksValue = System.getProperty(FALLBACKS_PROPERTY, FALLBACK_DEFAULTS);
88      final String[] fallbacks = FALLBACK_SEPARATOR.split(requestedFallbacksValue);
89      final ImmutableList.Builder<NameService> builder = ImmutableList.builder();
90      for (final String fallback : fallbacks) {
91        addFallback(fallback, builder);
92      }
93      FALLBACKS = builder.build();
94      if (FALLBACKS.isEmpty()) {
95        LOG.info("Joala DNS Fallbacks disabled.");
96      }
97    }
98  
99    /**
100    * <p>
101    * Add fallback as requested by id. Surrounding spaces and case is ignored. Unknown fallbacks are silently
102    * ignored so that you could for example specify "disabled" in order to have no fallbacks at all.
103    * </p>
104    *
105    * @param id      the id of the fallback; supported {@code dnsjava} and {@code default}
106    * @param builder builder to add the fallback instances to
107    */
108   private static void addFallback(@Nonnull final String id, @Nonnull final ImmutableList.Builder<NameService> builder) {
109     try {
110       NameService service = null;
111       if (FALLBACK_DNSJAVA_ID.equalsIgnoreCase(id.trim())) {
112         service = new DNSJavaNameServiceDescriptor().createNameService();
113       } else if (FALLBACK_DEFAULT_ID.equalsIgnoreCase(id.trim())) {
114         service = fallbackNameService();
115       }
116       if (service != null) {
117         builder.add(service);
118         LOG.info(format("Adding Name Service %s as fallback: %s", id, service.getClass()));
119       }
120     } catch (Exception e) {
121       LOG.warn(format("Unable to create fallback service %s. Fallback ignored.", id), e);
122     }
123   }

125   @Override
126   @Nonnull
127   public InetAddress[] lookupAllHostAddr(@Nonnull final String name) throws UnknownHostException {
128     checkNotNull(name, "Hostname must not be null.");
129     final InetAddress[] fromStore = NameStore.nameStore().lookup(name);
130     if (fromStore.length > 0) {
131       return fromStore;
132     }
133     return fallbackLookupAllHostAddr(name);
134   }
135 
136   @Override
137   @Nonnull
138   public String getHostByAddr(@Nonnull final byte[] addr) throws UnknownHostException {
139     checkNotNull(addr, "Address must not be null.");
140     final Name fromStore = NameStore.nameStore().reverseLookup(InetAddress.getByAddress(addr));
141     if (fromStore != null) {
142       return fromStore.toString();
143     }
144     return fallbackGetHostByAddr(addr);
145   }

147   /**
148    * <p>
149    * Go through fallbacks to resolve a given hostname.
150    * </p>
151    *
152    * @param name the hostname to resolve
153    * @return the address the hostname got resolved to
154    * @throws UnknownHostException if the address could not be resolved
155    */
156   @Nonnull
157   private InetAddress[] fallbackLookupAllHostAddr(@Nonnull final String name) throws UnknownHostException {
158     InetAddress[] fromFallbacks = null;
159     UnknownHostException firstException = null;
160     for (final NameService nameService : FALLBACKS) {
161       try {
162         fromFallbacks = nameService.lookupAllHostAddr(name);
163         break;
164       } catch (UnknownHostException e) {
165         if (firstException == null) {
166           firstException = e;
167         }
168       }
169     }
170     return handleFallbackResponse(
171             fromFallbacks,
172             firstException,
173             format("Could not find any fallback service which could resolve %s.", name));
174   }
175 
176   /**
177    * <p>
178    * Ask fallbacks for host by provided address.
179    * </p>
180    *
181    * @param addr the address to search for
182    * @return name of the host with the given address
183    * @throws UnknownHostException if such a host could not be found in the fallbacks
184    */
185   @Nonnull
186   private String fallbackGetHostByAddr(@Nonnull final byte[] addr) throws UnknownHostException {
187     String fromFallbacks = null;
188     UnknownHostException firstException = null;
189     for (final NameService nameService : FALLBACKS) {
190       try {
191         fromFallbacks = nameService.getHostByAddr(addr);
192         break;
193       } catch (UnknownHostException e) {
194         if (firstException == null) {
195           firstException = e;
196         }
197         fromFallbacks = null;
198       }
199     }
200     return handleFallbackResponse(
201             fromFallbacks,
202             firstException,
203             format("Could not find any fallback service which could resolve address %s.", Arrays.toString(addr)));
204   }
205 
206   /**
207    * <p>
208    * Handles the provided response from fallbacks.
209    * </p>
210    *
211    * @param fromFallbacks  the answer from the fallbacks; a value of {@code null} causes an {@link UnknownHostException}
212    * @param firstException the first exception provided by the fallbacks - if any; will be reused if it is required
213    *                       to throw an {@link UnknownHostException}.
214    * @param message        the message to add to exception if {@code firstException} is not used
215    * @param <T>            the response type
216    * @return either the response as provided by fallback or ends with exception
217    * @throws UnknownHostException exception if the {@code fromFallbacks} was {@code null}
218    */
219   @Nonnull
220   private <T> T handleFallbackResponse(@Nullable final T fromFallbacks,
221                                        @Nullable final UnknownHostException firstException,
222                                        @Nonnull final String message) throws UnknownHostException {
223     if (fromFallbacks == null) {
224       if (firstException == null) {
225         throw new UnknownHostException(message);
226       } else {
227         throw firstException;
228       }
229     }
230     return fromFallbacks;
231   }
