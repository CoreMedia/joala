56  /**
57   * <p>
58   * An webservice meant to be used in tests to simulate a HTTP Server. You can feed
59   * it with prepared responses which will then be sent on each request.
60   * </p>
61   *
62   * @since 10/4/12
63   */
64  public class EmbeddedWebservice {
65    /**
66     * Logging instance.
67     */
68    private static final Logger LOG = LoggerFactory.getLogger(EmbeddedWebservice.class);
69    private static final TimeoutImpl SERVER_STARTUP_TIMEOUT = new TimeoutImpl(5L, TimeUnit.SECONDS);
70  
71    /**
72     * Server to bind to port.
73     */
74    private volatile HttpServer server;
75    /**
76     * Http Handler which will provide prepared responses on each request.
77     */
78    @Nonnull
79    private final PreparedResponsesHttpHandler preparedResponsesHttpHandler;
80    /**
81     * The URI you can send requests to, which will reach this webservice.
82     */
83    @Nonnull
84    private final URI clientUri;
85    /**
86     * Context the service is bound to.
87     */
88    @Nonnull
89    private final String context;
90    /**
91     * Port the service is bound to.
92     */
93    private final int port;
94  
95    /**
96     * <p>
97     * Creates a webservice responding to a random free port on root context.
98     * </p>
99     *
100    * @throws IOException on failure
101    */
102   public EmbeddedWebservice() throws IOException {
103     this("/");
104   }
105 
106   /**
107    * <p>
108    * Creates a webservice responding to a random free port on specified context.
109    * </p>
110    *
111    * @param context context, for example {@code /lorem} for lorem-context
112    * @throws IOException on failure
113    */
114   public EmbeddedWebservice(@Nonnull final String context) throws IOException {
115     this(context, freePort());
116   }
117 
118   /**
119    * <p>
120    * Creates a webservice responding to given port on specified context.
121    * </p>
122    *
123    * @param context context, for example {@code /lorem} for lorem-context
124    * @param port    port to bind the service to
125    * @throws IOException on failure
126    */
127   public EmbeddedWebservice(@Nonnull final String context, final int port) throws IOException {
128     checkNotNull(context, "Context must not be null. For root context use '/'.");
129     try {
130       this.context = context;
131       this.port = port;
132       final InetSocketAddress address = new InetSocketAddress(port);
133       server = HttpServer.create(address, 0);
134       preparedResponsesHttpHandler = new PreparedResponsesHttpHandlerImpl();
135       server.createContext(this.context, preparedResponsesHttpHandler);
136       server.setExecutor(null);
137       clientUri = URI.create(format("http://%s:%d%s", getByName(null).getHostName(), this.port, context));
138     } catch (IOException e) {
139       throw new IOException("Failed to prepare embedded webservice for " + context + ':' + port, e);
140     }
141     LOG.info("Embedded Webservice ready to start for " + context + ':' + port);
142   }
143 
144   /**
145    * <p>
146    * Get the URI which where to send requests to, to reach this service.
147    * </p>
148    *
149    * @return client URI
150    */
151   @Nonnull
152   public URI getClientUri() {
153     return clientUri;
154   }
155 
156   /**
157    * Start the webservice.
158    */
159   public void start() {
160     checkState(server != null, "Server cannot be restarted.");
161     server.start();
162     preparedResponsesHttpHandler.feedResponses(statusCode(HttpURLConnection.HTTP_OK));
163     final DefaultHttpClient httpClient = getStartupHttpClient();
164     try {
165       getStartupCondition(httpClient).assumeThat(equalTo(HttpURLConnection.HTTP_OK));
166     } finally {
167       httpClient.getConnectionManager().shutdown();
168     }
169     LOG.info("Started embedded webservice at port {} with context {}.", port, context);
170   }
171 
172   private Condition<Integer> getStartupCondition(final DefaultHttpClient httpClient) {
173     return new DefaultConditionFactory(SERVER_STARTUP_TIMEOUT).condition(new ServerStartupStatusCodeExpression(httpClient));
174   }
175 
176   private DefaultHttpClient getStartupHttpClient() {
177     final DefaultHttpClient httpClient = new DefaultHttpClient();
178     final HttpParams httpParams = httpClient.getParams();
179     HttpConnectionParams.setConnectionTimeout(httpParams, (int) SERVER_STARTUP_TIMEOUT.in(TimeUnit.MILLISECONDS));
180     HttpConnectionParams.setSoTimeout(httpParams, (int) SERVER_STARTUP_TIMEOUT.in(TimeUnit.MILLISECONDS));
181     return httpClient;
182   }
183 
184   /**
185    * Stop the webservice.
186    */
187   public void stop() {
188     server.stop(1);
189     preparedResponsesHttpHandler.clearResponses();
190     server = null;
191     LOG.info("Stopped embedded webservice");
192   }
193 
194   /**
195    * <p>
196    * Get handler to control response flow.
197    * </p>
198    *
199    * @return handler to provide responses
200    */
201   @Nonnull
202   public PreparedResponsesHttpHandler getHttpHandler() {
203     return preparedResponsesHttpHandler;
204   }
205 
206   @Override
207   public String toString() {
208     return Objects.toStringHelper(this)
209             .add("clientUri", clientUri)
210             .add("server", server)
211             .add("preparedResponsesHttpHandler", preparedResponsesHttpHandler)
212             .add("context", context)
213             .add("port", port)
214             .toString();
215   }
216 
217   private class ServerStartupStatusCodeExpression extends AbstractExpression<Integer> {
218     private final DefaultHttpClient httpClient;
219 
220     private ServerStartupStatusCodeExpression(final DefaultHttpClient httpClient) {
221       this.httpClient = httpClient;
222     }
223 
224     @Override
225     public Integer get() {
226       final HttpUriRequest httpHead = new HttpHead(clientUri);
227       try {
228         final HttpResponse response = httpClient.execute(httpHead);
229         final HttpEntity httpEntity = response.getEntity();
230         final StatusLine statusLine = response.getStatusLine();
231         final int statusCode = statusLine.getStatusCode();
232         if (httpEntity != null) {
233           EntityUtils.consume(httpEntity);
234         }
235         return statusCode;
236       } catch (IOException e) {
237         throw new ExpressionEvaluationException(format("Failure reading from URI %s.", clientUri), e);
238       }
239     }
240   }
