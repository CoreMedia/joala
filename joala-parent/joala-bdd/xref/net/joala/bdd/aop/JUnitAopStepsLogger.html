41  /**
42   * <p>
43   * This logger is responsible to log JUnit-based methods using BDD-like naming conventions.
44   * </p>
45   * <p>
46   * The naming conventions for pointcuts are methods which match:
47   * </p>
48   * <ul>
49   * <li>{@code execution(* given_*(..))} or</li>
50   * <li>{@code execution(* when_*(..))} or</li>
51   * <li>{@code execution(* then_*(..))}</li>
52   * </ul>
53   * <p>
54   * To activate this steps logger, ensure that:
55   * </p>
56   * <ul>
57   * <li>your steps come from a Spring Bean (thus from a class injected to your test),</li>
58   * <li>if you use an internal class it must not be final and must be public,</li>
59   * <li>that you have a context configuration similar to:
60   * <pre>{@code
61   * <beans xmlns="http://www.springframework.org/schema/beans"
62   *        xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
63   *        xmlns:context="http://www.springframework.org/schema/context"
64   *        xsi:schemaLocation="http://www.springframework.org/schema/beans
65   *            http://www.springframework.org/schema/beans/spring-beans.xsd
66   *            http://www.springframework.org/schema/context
67   *            http://www.springframework.org/schema/context/spring-context.xsd">
68   *  <import resource="classpath:META-INF/joala/bdd/bdd-context.xml"/>
69   *  <context:component-scan base-package="your.test.package"/>
70   * </beans>
71   * }</pre>
72   * </li>
73   * </ul>
74   *
75   * @since 6/1/12
76   * @see net.joala.bdd
77   */
78  @SuppressWarnings({"JavaDoc", "ProhibitedExceptionDeclared", "ProhibitedExceptionThrown"})
79  @Aspect
80  public class JUnitAopStepsLogger {
81    private static final Logger LOG = LoggerFactory.getLogger(JUnitAopStepsLogger.class);
82  
83    /**
84     * <p>
85     * Adviser to log steps.
86     * </p>
87     *
88     * @param joinPoint where we are
89     * @return the result of the step call
90     * @throws Throwable in case of any error
91     */
92    @Around("execution(* given_*(..))||execution(* when_*(..))||execution(* then_*(..))")
93    public Object logGivenWhenThen(@Nonnull final ProceedingJoinPoint joinPoint) throws Throwable { // NOSONAR: Need to deal with generic throwables here
94      final String stepName = joinPoint.getSignature().getName();
95      final Object[] arguments = joinPoint.getArgs();
96      final Description stepDescription = describeStep(stepName, arguments);
97      final Object result;
98      try {
99        LOG.info("{}", stepDescription);
100       result = joinPoint.proceed();
101     } catch (Throwable throwable) { // NOSONAR: Need to deal with generic throwables here
102       LOG.info("{} (FAILED)", stepDescription);
103       throw throwable;
104     }
105     return result;
106   }
107 
108   /**
109    * Describe the step, either inserting argument descriptions
110    * into the placeholders of the step name or appending them.
111    * 
112    * @param stepName the step name
113    * @param arguments the arguments
114    * @return the description
115    */
116   private Description describeStep(final CharSequence stepName, final Object[] arguments) {
117     final StringBuilder text = new StringBuilder();
118     int pos = 0;
119     boolean placeholderFound = false;
120     while (pos < stepName.length()) {
121       final char c = stepName.charAt(pos++);
122       if (c == '$') {
123         int index = 0;
124         while (pos < stepName.length()) {
125           final int digit = Character.digit(stepName.charAt(pos++), 10);
126           if (digit == -1) {
127             pos --;
128             break;
129           }
130           index = index * 10 + digit;
131         }
132         if (index < arguments.length) {
133           final Object argument = arguments[index];
134           text.append(describeArgument(argument));
135           placeholderFound = true;
136         } else {
137           text.append('$').append(index);
138         }
139       } else if (c == '_') {
140         text.append(' ');
141       } else {
142         text.append(c);
143       }
144     }
145     
146     final Description stepDescription = new StringDescription();
147     stepDescription.appendText(text.toString().trim());
148     if (!placeholderFound) {
149       describeArguments(stepDescription, arguments);
150     }
151     return stepDescription;
152   }
153 
154   private Object describeArgument(final Object argument) {
155     if (argument == null) {
156       return "<null>";
157     } else if (argument instanceof String) {
158       return "\"" + argument + '"';
159     } else if (argument instanceof SelfDescribingReference) {
160       return '<' + ((SelfDescribingReference)argument).getName() + '>';
161     } else {
162       return argument.toString();
163     }
164   }
165 
166   /**
167    * <p>
168    * Describe the step arguments (most likely references) if they are describable. Ignore them if
169    * they are not self describable.
170    * </p>
171    *
172    * @param stepDescription description to add description of arguments to
173    * @param args            arguments to describe
174    */
175   private void describeArguments(@Nonnull final Description stepDescription, @Nonnull final Object[] args) {
176     final List<Object> argsList = Arrays.asList(args);
177     final Iterable<Object> descArgs = Iterables.filter(argsList, new IsSelfDescribing());
178     final boolean hasDescribableArgs = !Iterables.isEmpty(descArgs);
179     if (hasDescribableArgs) {
180       stepDescription.appendText(" (");
181       final Iterable<String> transformed = Iterables.transform(descArgs, new DescribeArgument());
182       stepDescription.appendText(Joiner.on(',').join(transformed));
183       stepDescription.appendText(")");
184     }
185   }
186 
187   /**
188    * <p>
189    * Filter to locate self describing arguments.
190    * </p>
191    */
192   private static class IsSelfDescribing implements Predicate<Object> {
193     @Override
194     public boolean apply(@Nullable final Object input) {
195       return input instanceof SelfDescribing || input instanceof SelfDescribing[];
196     }
197   }
198 
199   private static class DescribeArgument implements Function<Object, String> {
200     @Override
201     public String apply(@Nullable final Object input) {
202       final Description description = new StringDescription();
203       if (input instanceof SelfDescribing) {
204         description.appendDescriptionOf((SelfDescribing) input);
205       } else if (input instanceof SelfDescribing[]) {
206         description.appendList("{", ",", "}", Arrays.asList((SelfDescribing[]) input));
207       }
208       return description.toString();
209     }
210   }
