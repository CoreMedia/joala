358   private final class TestedDeceleratingWait extends DeceleratingWait {
359     private TestedDeceleratingWait() {
360     }
361 
362     private TestedDeceleratingWait(@Nonnull final Timeout timeout) {
363       super(timeout);
364     }
365 
366     private TestedDeceleratingWait(@Nonnull final Timeout timeout, @Nonnull final WaitFailStrategy failStrategy) {
367       super(timeout, failStrategy);
368     }
369 
370     private TestedDeceleratingWait(@Nonnull final Timeout timeout, @Nonnegative final double timeoutFactor) {
371       super(timeout, timeoutFactor);
372     }
373 
374     void until() {
375       this.until(RANDOM_STRING_SUPPLIER.get(), mockInput, stateQuery, Matchers.equalTo(Boolean.TRUE));
376     }
377 
378     @Override
379     protected long nowMillis() {
380       return nextCall(WaitCall.NOW).asLong();
381     }
382 
383     @Override
384     protected void sleep(final long millis) {
385       final ExpectedCall call = nextCall(WaitCall.SLEEP);
386       final long expected = call.asLong();
387       assertEquals(format("Expected number of milliseconds to wait (%s).", call), expected, millis);
388     }
389 
390   }

396   private class StateQueryFunction implements Function<Object, Boolean> {
397     @Override
398     public Boolean apply(@Nullable final Object input) {
399       return nextCall(WaitCall.FUNCTION_RESULT).asBoolean();
400     }
401   }
402 
403   private ExpectedCall nextCall(@Nonnull final WaitCall type) {
404     final ExpectedCall peek = expectedCalls.peek();
405     if (peek == null) {
406       fail(format("Call of type %s but did not expect any more calls.", type));
407     }
408     if (!type.equals(peek.getCall())) {
409       fail(format("Call of type %s but expected %s (%s).", type, peek.getCall(), peek));
410     }
411     final ExpectedCall result = expectedCalls.poll();
412     if (result.getResult() instanceof RuntimeException) {
413       throw (RuntimeException) result.getResult();
414     }
415     return result;
416   }

418   private static final class ExpectedCallsBuilder {
419     private final Collection<ExpectedCall> localExpectedCalls = new ArrayList<>();
420 
421     ExpectedCallsBuilder add(final WaitCall call, final Object result) {
422       localExpectedCalls.add(new ExpectedCall(localExpectedCalls.size(), call, result));
423       return this;
424     }
425 
426     LinkedList<ExpectedCall> build() {
427       return new LinkedList<>(localExpectedCalls);
428     }
429   }

431   private static final class ExpectedCall {
432     private final int position;
433     private final WaitCall call;
434     private final Object result;
435 
436     private ExpectedCall(final int position, final WaitCall call, final Object result) {
437       this.position = position;
438       this.call = call;
439       this.result = result;
440     }
441 
442     WaitCall getCall() {
443       return call;
444     }
445 
446     Object getResult() {
447       return result;
448     }
449 
450     long asLong() {
451       return (Long) result;
452     }
453 
454     @Override
455     public String toString() {
456       return MoreObjects.toStringHelper(this)
457                         .add("position", position)
458                         .add("call", call)
459                         .add("result", result)
460                         .toString();
461     }
462 
463     Boolean asBoolean() {
464       return (Boolean) result;
465     }
466   }
