46  /**
47   * <p>
48   * Base test for classes implementing {@link WaitFailStrategy}.
49   * </p>
50   *
51   * @since 8/26/12
52   */
53  @RunWith(MockitoJUnitRunner.class)
54  public abstract class WaitFailStrategyTest<S extends WaitFailStrategy, T extends Throwable> {
55    private static final RandomStringGenerator RANDOM_STRING_GENERATOR = new RandomStringGenerator.Builder().withinRange('a', 'z').build();
56    private static final PrimitiveIterator.OfLong LONGS = new Random().longs(0L, 1000L).iterator();
57  
58    private static final Supplier<String> FAIL_MESSAGE_PROVIDER = () -> "failMessage_" + RANDOM_STRING_GENERATOR.generate(20);
59    private static final Supplier<String> EXCEPTION_MESSAGE_PROVIDER = () -> "exceptionMessage_" + RANDOM_STRING_GENERATOR.generate(20);
60    private static final Supplier<Long> CONSUMED_MILLIS_PROVIDER = LONGS::next;
61  
62    @SuppressWarnings("UnusedDeclaration")
63    @Mock
64    private Object failedFunction;
65    @SuppressWarnings("UnusedDeclaration")
66    @Mock
67    private Object failedInput;
68    @SuppressWarnings("UnusedDeclaration")
69    @Mock
70    private Object failedLastValue;
71    @SuppressWarnings("UnusedDeclaration")
72    @Mock
73    private Throwable lastFailure;
74  
75    @Test
76    public void failed_match_method_should_throw_exception() throws Exception {
77      final S strategy = getFailStrategy();
78      boolean success = false;
79      try {
80        strategy.fail(FAIL_MESSAGE_PROVIDER.get(), failedFunction, failedInput, failedLastValue, nullValue(), CONSUMED_MILLIS_PROVIDER.get());
81        success = true;
82      } catch (RuntimeException ignored) {
83        // fine, that's what we expected
84      } catch (Error ignored) {
85        // fine that's what we expected
86      } catch (Throwable t) {
87        fail(format("Unexpected exception of type %s thrown.", t.getClass()));
88      }
89      assertFalse("An exception should have been thrown.", success);
90    }
91  
92    @Test
93    public void failed_with_exceptions_method_should_throw_exception() throws Exception {
94      final S strategy = getFailStrategy();
95      boolean success = false;
96      try {
97        strategy.fail(FAIL_MESSAGE_PROVIDER.get(), failedFunction, failedInput, lastFailure, CONSUMED_MILLIS_PROVIDER.get());
98        success = true;
99      } catch (RuntimeException ignored) {
100       // fine, that's what we expected
101     } catch (Error ignored) {
102       // fine that's what we expected
103     } catch (Throwable t) {
104       fail(format("Unexpected exception of type %s thrown.", t.getClass()));
105     }
106     assertFalse("An exception should have been thrown.", success);
107   }
108 
109   @Test
110   public void message_on_failed_match_should_be_contained_in_exception() throws Throwable {
111     final S strategy = getFailStrategy();
112     final String message = FAIL_MESSAGE_PROVIDER.get();
113     boolean success = false;
114     try {
115       strategy.fail(message, failedFunction, failedInput, failedLastValue, nullValue(), CONSUMED_MILLIS_PROVIDER.get());
116       success = true;
117     } catch (Throwable t) {
118       if (!getRaisedExceptionType().isAssignableFrom(t.getClass())) {
119         throw t;
120       }
121       assertThat("Message should be contained in exception message.", t.getMessage(), containsString(message));
122     }
123     assertFalse("An exception should have been thrown.", success);
124   }
125 
126   @Test
127   public void message_when_failed_with_exception_should_be_contained_in_exception() throws Throwable {
128     final S strategy = getFailStrategy();
129     boolean success = false;
130     final String message = FAIL_MESSAGE_PROVIDER.get();
131     try {
132       strategy.fail(message, failedFunction, failedInput, lastFailure, CONSUMED_MILLIS_PROVIDER.get());
133       success = true;
134     } catch (Throwable t) {
135       if (!getRaisedExceptionType().isAssignableFrom(t.getClass())) {
136         throw t;
137       }
138       assertThat("Message should be contained in exception message.", t.getMessage(), containsString(message));
139     }
140     assertFalse("An exception should have been thrown.", success);
141   }
142 
143   @Test
144   public void cause_when_failed_with_exception_should_be_contained_in_exception() throws Throwable {
145     final S strategy = getFailStrategy();
146     final String evaluationExceptionMessage = EXCEPTION_MESSAGE_PROVIDER.get();
147     doAnswer(invocation -> {
148       invocation.getArgumentAt(0, PrintWriter.class).append(evaluationExceptionMessage);
149       return null;
150     }).when(lastFailure).printStackTrace(any(PrintWriter.class));
151     boolean success = false;
152     try {
153       strategy.fail(FAIL_MESSAGE_PROVIDER.get(), failedFunction, failedInput, lastFailure, CONSUMED_MILLIS_PROVIDER.get());
154       success = true;
155     } catch (Throwable t) {
156       if (!getRaisedExceptionType().isAssignableFrom(t.getClass())) {
157         throw t;
158       }
159       assertThat("Cause should be mentioned in failure.", t,
160               anyOf(
161                       causedBy(lastFailure),
162                       messageContains(evaluationExceptionMessage, true)
163               )
164       );
165     }
166     assertFalse("An exception should have been thrown.", success);
167   }
168 
169   @Test
170   public void should_throw_expected_exception_type_on_failure_because_of_exception() throws Exception {
171     final S strategy = getFailStrategy();
172     try {
173       strategy.fail(FAIL_MESSAGE_PROVIDER.get(), failedFunction, failedInput, lastFailure, CONSUMED_MILLIS_PROVIDER.get());
174       fail("Exception should have been thrown.");
175     } catch (Throwable t) {
176       assertThat("Raised exception should be of expected type.", t, Matchers.instanceOf(getRaisedExceptionType()));
177     }
178   }
179 
180   @Test
181   public void should_throw_expected_exception_type_on_failure_because_mismatch() throws Exception {
182     final S strategy = getFailStrategy();
183     try {
184       strategy.fail(FAIL_MESSAGE_PROVIDER.get(), failedFunction, failedInput, failedLastValue, nullValue(), CONSUMED_MILLIS_PROVIDER.get());
185       fail("Exception should have been thrown.");
186     } catch (Throwable t) {
187       assertThat("Raised exception should be of expected type.", t, Matchers.instanceOf(getRaisedExceptionType()));
188     }
189   }

193 
194   protected abstract S getFailStrategy();
195 }
