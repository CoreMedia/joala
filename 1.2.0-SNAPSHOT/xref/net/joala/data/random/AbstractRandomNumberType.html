20  package net.joala.data.random;
21  
22  import com.google.common.base.MoreObjects;
23  import com.google.common.collect.Range;
24  
25  import javax.annotation.Nonnull;
26  
27  import static com.google.common.base.Preconditions.checkArgument;
28  import static com.google.common.base.Preconditions.checkNotNull;
29  
30  /**
31   * <p>
32   * Abstract implementation of RandomNumberType holding the type of numbers to create.
33   * </p>
34   *
35   * @param <T> number type to generate
36   * @deprecated Will be removed soon.
37   */
38  @SuppressWarnings("deprecation")
39  @Deprecated
40  public abstract class AbstractRandomNumberType<T extends Comparable<? extends Number>> implements RandomNumberType<T> {
41    private static final Range<Double> VALID_PERCENTAGE_RANGE = Range.closed(0d, 1d);
42    /**
43     * Type of numbers to create.
44     */
45    @Nonnull
46    private final Class<T> type;
47  
48    /**
49     * <p>
50     * Constructor specifying type of random numbers.
51     * </p>
52     *
53     * @param type type of numbers
54     */
55    protected AbstractRandomNumberType(@Nonnull final Class<T> type) {
56      checkNotNull(type, "Type must not be null.");
57      this.type = type;
58    }
59  
60    @Override
61    @Nonnull
62    public final Class<T> getType() {
63      return type;
64    }
65  
66    @Override
67    public String toString() {
68      return MoreObjects.toStringHelper(this)
69              .add("type", type)
70              .toString();
71    }
72  
73    /**
74     * Verifies that the given percentage is between 0 and 1.
75     *
76     * @param percent percentage value to check
77     * @throws IllegalArgumentException if percentage is out of the valid bounds
78     */
79    protected void checkPercentageArgument(final double percent) {
80      checkArgument(VALID_PERCENTAGE_RANGE.contains(percent),
81              "Given percentage must be between 0 and 1 but is %s.",
82              percent);
83    }
84  }
